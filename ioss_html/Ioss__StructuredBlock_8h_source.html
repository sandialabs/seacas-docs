130  /** \brief Set the 'offset' for the block.
131  *
132  * The 'offset' is used to map a cell or node location within a
133  * structured block to the model implicit cell or node location
134  * on a single processor. zero-based.
135  *
136  * The 'global' offsets do the same except for they apply over
137  * the entire model on all processors. zero-based.
138  *
139  * For example, the file descriptor (1-based) of
140  * the 37th cell in the 4th block is calculated by:
141  *
142  * file_descriptor = offset of block 4 + 37
143  *
144  * This can also be used to determine which structured block
145  * a cell with a file_descriptor maps into. An particular
146  * structured block contains all cells in the range:
147  *
148  * offset < file_descriptor <= offset+number_cells_per_block
149  *
150  * Note that for nodes, the nodeOffset does not take into account
151  * the nodes that are shared between blocks.
152  */
153  void set_node_offset(size_t offset) { m_nodeOffset = offset; }
154  void set_cell_offset(size_t offset) { m_cellOffset = offset; }
155  void set_node_global_offset(size_t offset) { m_nodeGlobalOffset = offset; }
156  void set_cell_global_offset(size_t offset) { m_cellGlobalOffset = offset; }
157 
158  size_t get_node_offset() const { return m_nodeOffset; }

227  template <typename INT_t> size_t get_cell_node_ids(INT_t *idata, bool add_offset) const
228  {
229  // Fill 'idata' with the cell node ids which are the
230  // 1-based location of each node in this zone
231  // The location is based on the "model" (all processors) zone.
232  // If this is a parallel decomposed model, then
233  // this block may be a subset of the "model" zone
234  //
235  // if 'add_offset' is true, then add the m_cellGlobalOffset
236  // which changes the location to be the location in the
237  // entire "mesh" instead of within a "zone" (all processors)
238 
239  size_t index = 0;
240  size_t offset = add_offset ? m_nodeGlobalOffset : 0;
241 
242  if (m_nk == 0 && m_nj == 0 && m_ni == 0) {
243  return index;
244  }
245 
246  for (int kk = 0; kk < m_nk + 1; kk++) {
247  size_t k = m_offsetK + kk;
248  for (int jj = 0; jj < m_nj + 1; jj++) {
249  size_t j = m_offsetJ + jj;
250  for (int ii = 0; ii < m_ni + 1; ii++) {
251  size_t i = m_offsetI + ii;
252 
253  size_t ind = k * (m_niGlobal + 1) * (m_njGlobal + 1) + j * (m_niGlobal + 1) + i;
254 
255  idata[index++] = ind + offset + 1;
256  }
257  }
258  }
259 
260  for (auto idx_id : m_globalIdMap) {
261  idata[idx_id.first] = idx_id.second;
262  }
263 
264  return index;
265  }
266 
267  template <typename INT_t> size_t get_cell_ids(INT_t *idata, bool add_offset) const
268  {
269  // Fill 'idata' with the cell ids which are the
270  // 1-based location of each cell in this zone
271  // The location is based on the "model" zone.
272  // If this is a parallel decomposed model, then
273  // this block may be a subset of the "model" zone
274  //
275  // if 'add_offset' is true, then add the m_cellGlobalOffset
276  // which changes the location to be the location in the
277  // entire "mesh" instead of within a "zone"
278 
279  size_t index = 0;
280  size_t offset = add_offset ? m_cellGlobalOffset : 0;
281 
282  if (m_nk == 0 && m_nj == 0 && m_ni == 0) {
283  return index;
284  }
285 
286  for (int kk = 0; kk < m_nk; kk++) {
287  size_t k = m_offsetK + kk;
288  for (int jj = 0; jj < m_nj; jj++) {
289  size_t j = m_offsetJ + jj;
290  for (int ii = 0; ii < m_ni; ii++) {
291  size_t i = m_offsetI + ii;
292 
293  size_t ind = k * m_niGlobal * m_njGlobal + j * m_niGlobal + i;
294 
295  idata[index++] = ind + offset + 1;
296  }
297  }
298  }
299  return index;
300  }
