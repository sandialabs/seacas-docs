142 /** \brief Set the 'offset' for the block.
143 *
144 * The 'offset' is used to map a cell or node location within a
145 * structured block to the model implicit cell or node location
146 * on a single processor. zero-based.
147 *
148 * The 'global' offsets do the same except for they apply over
149 * the entire model on all processors. zero-based.
150 *
151 * For example, the file descriptor (1-based) of
152 * the 37th cell in the 4th block is calculated by:
153 *
154 * file_descriptor = offset of block 4 + 37
155 *
156 * This can also be used to determine which structured block
157 * a cell with a file_descriptor maps into. An particular
158 * structured block contains all cells in the range:
159 *
160 * offset < file_descriptor <= offset+number_cells_per_block
161 *
162 * Note that for nodes, the nodeOffset does not take into account
163 * the nodes that are shared between blocks.
164 */

166 void set_cell_offset(size_t offset) { m_cellOffset = offset; }

213 // Get the local (relative to this block on this processor) node id at the specified
214 // i,j,k location (1 <= i,j,k <= ni+1,nj+1,nk+1). 0-based.
215 size_t get_block_local_node_offset(int ii, int jj, int kk) const
216 {
217 auto i = ii - m_offset[0];
218 auto j = jj - m_offset[1];
219 auto k = kk - m_offset[2];
220 assert(i > 0 && i <= m_ijk[0] + 1 && j > 0 && j <= m_ijk[1] + 1 && k > 0 &&
221 k <= m_ijk[2] + 1);
222 return static_cast<size_t>(k - 1) * (m_ijk[0] + 1) * (m_ijk[1] + 1) +
223 static_cast<size_t>(j - 1) * (m_ijk[0] + 1) + i - 1;
224 }

251 template <typename INT_t> size_t get_cell_node_ids(INT_t *idata, bool add_offset) const
252 {
253 // Fill 'idata' with the cell node ids which are the
254 // 1-based location of each node in this zone
255 // The location is based on the "model" (all processors) zone.
256 // If this is a parallel decomposed model, then
257 // this block may be a subset of the "model" zone
258 //
259 // if 'add_offset' is true, then add the m_cellGlobalOffset
260 // which changes the location to be the location in the
261 // entire "mesh" instead of within a "zone" (all processors)
262
263 size_t index = 0;
264 size_t offset = add_offset ? m_nodeGlobalOffset : 0;
265
266 if (m_ijk[2] == 0 && m_ijk[1] == 0 && m_ijk[0] == 0) {
267 return index;
268 }
269
270 for (int kk = 0; kk < m_ijk[2] + 1; kk++) {
271 size_t k = m_offset[2] + kk;
272 for (int jj = 0; jj < m_ijk[1] + 1; jj++) {
273 size_t j = m_offset[1] + jj;
274 for (int ii = 0; ii < m_ijk[0] + 1; ii++) {
275 size_t i = m_offset[0] + ii;
276
277 size_t ind =
278 k * (m_ijkGlobal[0] + 1) * (m_ijkGlobal[1] + 1) + j * (m_ijkGlobal[0] + 1) + i;
279
280 idata[index++] = ind + offset + 1;
281 }
282 }
283 }
284
285 for (const auto &idx_id : m_globalIdMap) {
286 idata[idx_id.first] = idx_id.second;
287 }
288
289 return index;
290 }

292 template <typename INT_t> size_t get_cell_ids(INT_t *idata, bool add_offset) const
293 {
294 // Fill 'idata' with the cell ids which are the
295 // 1-based location of each cell in this zone
296 // The location is based on the "model" zone.
297 // If this is a parallel decomposed model, then
298 // this block may be a subset of the "model" zone
299 //
300 // if 'add_offset' is true, then add the m_cellGlobalOffset
301 // which changes the location to be the location in the
302 // entire "mesh" instead of within a "zone"
303
304 size_t index = 0;
305 size_t offset = add_offset ? m_cellGlobalOffset : 0;
306
307 if (m_ijk[2] == 0 && m_ijk[1] == 0 && m_ijk[0] == 0) {
308 return index;
309 }
310
311 for (int kk = 0; kk < m_ijk[2]; kk++) {
312 size_t k = m_offset[2] + kk;
313 for (int jj = 0; jj < m_ijk[1]; jj++) {
314 size_t j = m_offset[1] + jj;
315 for (int ii = 0; ii < m_ijk[0]; ii++) {
316 size_t i = m_offset[0] + ii;
317
318 size_t ind = k * m_ijkGlobal[0] * m_ijkGlobal[1] + j * m_ijkGlobal[0] + i;
319
320 idata[index++] = ind + offset + 1;
321 }
322 }
323 }
324 return index;
325 }
