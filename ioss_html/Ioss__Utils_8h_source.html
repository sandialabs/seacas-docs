135 template <typename T> static void generate_index(std::vector<T> &index)
136 {
137 T sum = 0;
138 for (size_t i = 0; i < index.size() - 1; i++) {
139 T cnt = index[i];
140 index[i] = sum;
141 sum += cnt;
142 }
143 index.back() = sum;
144 }

146 template <typename T> static T find_index_location(T node, const std::vector<T> &index)
147 {
148 // 0-based node numbering
149 // index[p] = first node (0-based) on processor p
150
151#if 1
152 // Assume data coherence. I.e., a new search will be close to the
153 // previous search.
154 static size_t prev = 1;
155
156 size_t nproc = index.size();
157 if (prev < nproc && index[prev - 1] <= node && index[prev] > node) {
158 return prev - 1;
159 }
160
161 for (size_t p = 1; p < nproc; p++) {
162 if (index[p] > node) {
163 prev = p;
164 return p - 1;
165 }
166 }
167 std::ostringstream errmsg;
168 errmsg << "FATAL ERROR: find_index_location. Searching for " << node << " in:\n";
169 for (auto idx : index) {
170 errmsg << idx << ", ";
171 }
172 errmsg << "\n";
173 IOSS_ERROR(errmsg);
174#else
175 return std::distance(index.begin(), std::upper_bound(index.begin(), index.end(), node)) - 1;
176#endif
177 }

204 /**
205 * Returns the number of digits required to print the number.
206 * If `use_commas` is specified, then the width will be adjusted
207 * to account for the comma used every 3 digits.
208 * (1,234,567,890 would return 13)
209 * Typically used with the `fmt::print()` functions as:
210 * ```
211 * fmt::print("{:{}}", number, number_width(number,true))
212 * fmt::print("{:{}d}", number, number_width(number,false))
213 * ```
214 */
215 inline static int number_width(const size_t number, bool use_commas = false)
216 {
217 if (number == 0) {
218 return 1;
219 }
220 int width = int(std::floor(std::log10(number))) + 1;
221 if (use_commas) {
222 width += ((width - 1) / 3);
223 }
224 return width;
225 }

227 inline static int power_2(int count)
228 {
229 // Return the power of two which is equal to or greater than `count`
230 // count = 15 -> returns 16
231 // count = 16 -> returns 16
232 // count = 17 -> returns 32
233
234 // Use brute force...
235 int pow2 = 1;
236 while (pow2 < count) {
237 pow2 *= 2;
238 }
239 return pow2;
240 }

525 // SEE: http://lemire.me/blog/2017/04/10/removing-duplicates-from-lists-quickly
526 template <typename T> static size_t unique(std::vector<T> &out, bool skip_first)
527 {
528 if (out.empty())
529 return 0;
530 size_t i = 1;
531 size_t pos = 1;
532 T oldv = out[0];
533 if (skip_first) {
534 i = 2;
535 pos = 2;
536 oldv = out[1];
537 }
538 for (; i < out.size(); ++i) {
539 T newv = out[i];
540 out[pos] = newv;
541 pos += (newv != oldv);
542 oldv = newv;
543 }
544 return pos;
545 }
