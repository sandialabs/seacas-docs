135 // allows finding if a value is of type 'const char *'
136 // and is null; if so, we cannot print it to an ostream
137 // directly!!!
138 template <class T> struct is_null_finder
139 {
140 bool is(const T & /*unused*/) const { return false; }
141 };
142
143 template <> struct is_null_finder<char *>
144 {
145 bool is(char *const &val) { return val == nullptr; }
146 };
147
148 template <> struct is_null_finder<const char *>
149 {
150 bool is(const char *const &val) { return val == nullptr; }
151 };

156struct Assert
157{
158 using assert_func = SmartAssert::assert_func;
159
160 // helpers, in order to be able to compile the code
161 Assert &SMART_ASSERT_A;
162 Assert &SMART_ASSERT_B;
163
164 explicit Assert(const char *expr)
165 : SMART_ASSERT_A(*this), SMART_ASSERT_B(*this), needs_handling_(true)
166 {
167 context_.set_expr(expr);
168
169 if ((logger() == nullptr) || handlers().size() < 4) {
170 // used before main!
171 Ioss::Private::init_assert();
172 }
173 }
174
175 Assert(const Assert &other)
176 : SMART_ASSERT_A(*this), SMART_ASSERT_B(*this), context_(other.context_),
177 needs_handling_(true)
178 {
179 other.needs_handling_ = false;
180 }
181
182 ~Assert()
183 {
184 if (needs_handling_) {
185 handle_assert();
186 }
187 }
188
189 template <class type> Assert &print_current_val(const type &val, const char *my_msg)
190 {
191 std::ostringstream out;
192
193 Ioss::Private::is_null_finder<type> f;
194 bool bIsNull = f.is(val);
195 if (!bIsNull) {
196 out << val;
197 }
198 else {
199 // null string
200 out << "null";
201 }
202 context_.add_val(out.str(), my_msg);
203 return *this;
204 }
205
206 Assert &print_context(const char *file, int line)
207 {
208 context_.set_file_line(file, line);
209 return *this;
210 }
211
212 Assert &msg(const char *strMsg)
213 {
214 context_.set_level_msg(strMsg);
215 return *this;
216 }
217
218 Assert &level(int nLevel, const char *strMsg = nullptr)
219 {
220 context_.set_level(nLevel);
221 context_.set_level_msg(strMsg);
222 return *this;
223 }
224
225 Assert &warn(const char *strMsg = nullptr) { return level(lvl_warn, strMsg); }
226
227 Assert &debug(const char *strMsg = nullptr) { return level(lvl_debug, strMsg); }
228
229 Assert &error(const char *strMsg = nullptr) { return level(lvl_error, strMsg); }
230
231 Assert &fatal(const char *strMsg = nullptr) { return level(lvl_fatal, strMsg); }

233 // in this case, we set the default logger, and make it
234 // write everything to this file
235 static void set_log(const char *strFileName)
236 {
237 Ioss::Private::set_default_log_name(strFileName);
238 logger() = &SmartAssert::default_logger;
239 }
240
241 // in this case, we set the default logger, and make it
242 // write everything to this log
243 static void set_log(std::ostream &out)
244 {
245 Ioss::Private::set_default_log_stream(out);
246 logger() = &SmartAssert::default_logger;
247 }
248
249 static void set_log(assert_func log) { logger() = log; }
250
251 static void set_handler(int nLevel, assert_func handler) { handlers()[nLevel] = handler; }
252
253private:
254 // handles the current assertion.
255 void handle_assert()
256 {
257 logger()(context_);
258 get_handler(context_.get_level())(context_);
259 }
260
261 /*
262 IMPORTANT NOTE:
263 The only reason logger & handlers are functions, are
264 because you might use SMART_ASSERT before main().
265
266 In this case, since they're statics, they might not
267 be initialized. However, making them functions
268 will make it work.
269 */
270
271 // the log
272 static assert_func &logger()
273 {
274 static assert_func inst;
275 return inst;
276 }
277
278 // the handler
279 using handlers_collection = std::map<int, assert_func>;
280 static handlers_collection &handlers()
281 {
282 static handlers_collection inst;
283 return inst;
284 }
285
286 static assert_func get_handler(int nLevel)
287 {
288 handlers_collection::const_iterator found = handlers().find(nLevel);
289 if (found != handlers().end()) {
290 return (*found).second;
291 }
292
293 // we always assume the debug handler has been set
294 return (*handlers().find(lvl_debug)).second;
295 }
296
297private:
298 assert_context context_;
299 mutable bool needs_handling_;
300};
