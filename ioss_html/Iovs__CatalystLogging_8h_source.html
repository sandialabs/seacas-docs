19 // Enables Catalyst log output from IOSS when Catalyst CGNS or
20 // Exodus IOSS databases are created. Rank 0 of the application
21 // writes an entry to a CSV (comma separted value) log file upon
22 // IOSS database creation. Log output is controlled through
23 // IOSS properties. IOSS properties that control logging must be
24 // prepened with the string "CATALYST_LOGGING_".
25 //
26 // There are three reserved IOSS property names for logging.
27 //
28 // "CATALYST_LOGGING_ENABLED" : enables Catalyst log output when passed
29 // an argument of "true or a non-zero integer". Default is "false".
30 //
31 // "CATALYST_LOGGING_FILE_NAME" : string specifying the log file output
32 // name. Default is "catalyst_log.csv".
33 //
34 // "CATALYST_LOGGING_OUTPUT_DIRECTORY_PATH" : string specifying an absolute
35 // or relative path to the log file. Default is current working directory, "".
36 //
37 // Real, Integer, and String IOSS properties prepended by the string
38 // "CATALYST_LOGGING_" are used to specify output for the log line in the CSV
39 // output log file. The logger will remove the prepended string and create
40 // a header line in the CSV file sorted by name.
41 //
42 // Example application code creating a Catalyst IOSS database with logging:
43 //
44 // Ioss::PropertyManager *p;
45 // p->add(Ioss::Property("CATALYST_LOGGING_ENABLED", true));
46 // p->add(Ioss::Property("CATALYST_LOGGING_FILE_NAME", "app_log.csv"));
47 // p->add(Ioss::Property("CATALYST_LOGGING_OUTPUT_DIRECTORY_PATH, "/etc/logs/"));
48 // p->add(Ioss::Property("CATALYST_LOGGING_NUM_RANKS", getNumRanks()));
49 // p->add(Ioss::Property("CATALYST_LOGGING_DATE", getDate()));
50 // p->add(Ioss::Property("CATALYST_LOGGING_USER", getUser()));
51 // p->add(Ioss::Property("CATALYST_LOGGING_APP_NAME", getAppName()));
52 //
53 // The IOSS properties contained in *p are passed to the IOSS
54 // Ioss::IOFactory::create(). A log entry will be appended to the log file
55 // at "/etc/logs/app_log_.csv" every time a Catalyst database is created by
56 // a running instance of the application. The CSV log file will have the following
57 // format, which can be easily read by Python.
58 //
59 // APP_NAME,DATE,NUM_RANKS,USER
60 // goo,November 10th,16,joe
61 // bar,December 12th,8,stan
62 //
63 // If an IOSS string property contains commas, these need to be quoted.
64 //
65 // p->add(Ioss::Property("CATALYST_LOGGING_ADDRESS", "\"123 main, PO 4, ND, 34422\""))
66 //
67 // Quotes inside strings must also be quoted.
68 //
69 // p->add(Ioss::Property("CATALYST_LOGGING_QUOTE", "I am \"\"Dave\"\""))
