75 /**
76 * Grow the hash table by a factor of GrowthFactor keeping the bucket count to a
77 * power of two. It allows the table to use a mask operation instead of a modulo
78 * operation to map a hash to a bucket.
79 *
80 * GrowthFactor must be a power of two >= 2.
81 */
82 template <std::size_t GrowthFactor> class power_of_two_growth_policy
83 {
84 public:
85 /**
86 * Called on the hash table creation and on rehash. The number of buckets for
87 * the table is passed in parameter. This number is a minimum, the policy may
88 * update this value with a higher value if needed (but not lower).
89 *
90 * If 0 is given, min_bucket_count_in_out must still be 0 after the policy
91 * creation and bucket_for_hash must always return 0 in this case.
92 */
93 explicit power_of_two_growth_policy(std::size_t &min_bucket_count_in_out)
94 {
95 if (min_bucket_count_in_out > max_bucket_count()) {
96 TSL_HH_THROW_OR_TERMINATE(std::length_error, "The hash table exceeds its maximum size.");
97 }
98
99 if (min_bucket_count_in_out > 0) {
100 min_bucket_count_in_out = round_up_to_power_of_two(min_bucket_count_in_out);
101 m_mask = min_bucket_count_in_out - 1;
102 }
103 else {
104 m_mask = 0;
105 }
106 }
107
108 /**
109 * Return the bucket [0, bucket_count()) to which the hash belongs.
110 * If bucket_count() is 0, it must always return 0.
111 */
112 std::size_t bucket_for_hash(std::size_t hash) const noexcept { return hash & m_mask; }
113
114 /**
115 * Return the bucket count to use when the bucket array grows on rehash.
116 */
117 std::size_t next_bucket_count() const
118 {
119 if ((m_mask + 1) > max_bucket_count() / GrowthFactor) {
120 TSL_HH_THROW_OR_TERMINATE(std::length_error, "The hash table exceeds its maximum size.");
121 }
122
123 return (m_mask + 1) * GrowthFactor;
124 }
125
126 /**
127 * Return the maximum number of buckets supported by the policy.
128 */
129 std::size_t max_bucket_count() const
130 {
131 // Largest power of two.
132 return (std::numeric_limits<std::size_t>::max() / 2) + 1;
133 }
134
135 /**
136 * Reset the growth policy as if it was created with a bucket count of 0.
137 * After a clear, the policy must always return 0 when bucket_for_hash is
138 * called.
139 */
140 void clear() noexcept { m_mask = 0; }
141
142 private:
143 static std::size_t round_up_to_power_of_two(std::size_t value)
144 {
145 if (is_power_of_two(value)) {
146 return value;
147 }
148
149 if (value == 0) {
150 return 1;
151 }
152
153 --value;
154 for (std::size_t i = 1; i < sizeof(std::size_t) * CHAR_BIT; i *= 2) {
155 value |= value >> i;
156 }
157
158 return value + 1;
159 }
160
161 static constexpr bool is_power_of_two(std::size_t value)
162 {
163 return value != 0 && (value & (value - 1)) == 0;
164 }
165
166 private:
167 static_assert(is_power_of_two(GrowthFactor) && GrowthFactor >= 2,
168 "GrowthFactor must be a power of two >= 2.");
169
170 std::size_t m_mask;
171 };

173 /**
174 * Grow the hash table by GrowthFactor::num / GrowthFactor::den and use a modulo
175 * to map a hash to a bucket. Slower but it can be useful if you want a slower
176 * growth.
177 */
178 template <class GrowthFactor = std::ratio<3, 2>> class mod_growth_policy
179 {
180 public:
181 explicit mod_growth_policy(std::size_t &min_bucket_count_in_out)
182 {
183 if (min_bucket_count_in_out > max_bucket_count()) {
184 TSL_HH_THROW_OR_TERMINATE(std::length_error, "The hash table exceeds its maximum size.");
185 }
186
187 if (min_bucket_count_in_out > 0) {
188 m_mod = min_bucket_count_in_out;
189 }
190 else {
191 m_mod = 1;
192 }
193 }
194
195 std::size_t bucket_for_hash(std::size_t hash) const noexcept { return hash % m_mod; }
196
197 std::size_t next_bucket_count() const
198 {
199 if (m_mod == max_bucket_count()) {
200 TSL_HH_THROW_OR_TERMINATE(std::length_error, "The hash table exceeds its maximum size.");
201 }
202
203 const double next_bucket_count =
204 std::ceil(double(m_mod) * REHASH_SIZE_MULTIPLICATION_FACTOR);
205 if (!std::isnormal(next_bucket_count)) {
206 TSL_HH_THROW_OR_TERMINATE(std::length_error, "The hash table exceeds its maximum size.");
207 }
208
209 if (next_bucket_count > double(max_bucket_count())) {
210 return max_bucket_count();
211 }
212 else {
213 return std::size_t(next_bucket_count);
214 }
215 }
216
217 std::size_t max_bucket_count() const { return MAX_BUCKET_COUNT; }
218
219 void clear() noexcept { m_mod = 1; }
220
221 private:
222 static constexpr double REHASH_SIZE_MULTIPLICATION_FACTOR =
223 1.0 * GrowthFactor::num / GrowthFactor::den;
224 static const std::size_t MAX_BUCKET_COUNT = std::size_t(
225 double(std::numeric_limits<std::size_t>::max() / REHASH_SIZE_MULTIPLICATION_FACTOR));
226
227 static_assert(REHASH_SIZE_MULTIPLICATION_FACTOR >= 1.1, "Growth factor should be >= 1.1.");
228
229 std::size_t m_mod;
230 };

326 /**
327 * Grow the hash table by using prime numbers as bucket count. Slower than
328 * tsl::hh::power_of_two_growth_policy in general but will probably distribute
329 * the values around better in the buckets with a poor hash function.
330 *
331 * To allow the compiler to optimize the modulo operation, a lookup table is
332 * used with constant primes numbers.
333 *
334 * With a switch the code would look like:
335 * \code
336 * switch(iprime) { // iprime is the current prime of the hash table
337 * case 0: hash % 5ul;
338 * break;
339 * case 1: hash % 17ul;
340 * break;
341 * case 2: hash % 29ul;
342 * break;
343 * ...
344 * }
345 * \endcode
346 *
347 * Due to the constant variable in the modulo the compiler is able to optimize
348 * the operation by a series of multiplications, substractions and shifts.
349 *
350 * The 'hash % 5' could become something like 'hash - (hash * 0xCCCCCCCD) >> 34)
351 * * 5' in a 64 bits environment.
352 */
353 class prime_growth_policy
354 {
355 public:
356 explicit prime_growth_policy(std::size_t &min_bucket_count_in_out)
357 {
358 auto it_prime =
359 std::lower_bound(detail::PRIMES.begin(), detail::PRIMES.end(), min_bucket_count_in_out);
360 if (it_prime == detail::PRIMES.end()) {
361 TSL_HH_THROW_OR_TERMINATE(std::length_error, "The hash table exceeds its maximum size.");
362 }
363
364 m_iprime = static_cast<unsigned int>(std::distance(detail::PRIMES.begin(), it_prime));
365 if (min_bucket_count_in_out > 0) {
366 min_bucket_count_in_out = *it_prime;
367 }
368 else {
369 min_bucket_count_in_out = 0;
370 }
371 }
372
373 std::size_t bucket_for_hash(std::size_t hash) const noexcept
374 {
375 return detail::MOD_PRIME[m_iprime](hash);
376 }
377
378 std::size_t next_bucket_count() const
379 {
380 if (m_iprime + 1 >= detail::PRIMES.size()) {
381 TSL_HH_THROW_OR_TERMINATE(std::length_error, "The hash table exceeds its maximum size.");
382 }
383
384 return detail::PRIMES[m_iprime + 1];
385 }
386
387 std::size_t max_bucket_count() const { return detail::PRIMES.back(); }
388
389 void clear() noexcept { m_iprime = 0; }
390
391 private:
392 unsigned int m_iprime;
393
394 static_assert(std::numeric_limits<decltype(m_iprime)>::max() >= detail::PRIMES.size(),
395 "The type of m_iprime is not big enough.");
396 };
