56 class PartIdMapping
57 {
58 public:
59 PartIdMapping() : m_idsAssigned(false)
60 {
61 set_error_handler([](const std::ostringstream &errmsg) { default_error_handler(errmsg); });
62 }
63
64 void register_part_name(const std::string &name)
65 {
66 m_partNames.push_back(name);
67 handle_block_part(name);
68 }
69
70 void register_part_name_with_id(const std::string &name, unsigned id)
71 {
72 register_part_name(name);
73 assign(name, id);
74 }
75
76 unsigned get(const std::string &name) const
77 {
78 if (!m_idsAssigned)
79 assign_ids();
80 return get_part_id(name);
81 }
82
83 std::string get(unsigned id) const
84 {
85 if (!m_idsAssigned)
86 assign_ids();
87 return get_part_name(id);
88 }
89
90 unsigned size() const
91 {
92 if (!m_idsAssigned)
93 assign_ids();
94 return m_ids.size();
95 }
96
97 std::vector<std::string> get_part_names_sorted_by_id() const
98 {
99 if (!m_idsAssigned)
100 assign_ids();
101
102 std::vector<std::string> names;
103 names.reserve(m_parts.size());
104
105 for (const auto &iter : m_parts) {
106 names.push_back(iter.second);
107 }
108
109 return names;
110 }
111
112 bool is_registered(const std::string &name) const { return m_ids.count(name) > 0; }
113
114 const std::vector<std::string> &get_part_names() const { return m_partNames; }
115
116 void set_error_handler(ErrorHandler errorHandler) { m_errorHandler = errorHandler; }
117
118 const std::string get_group_type() const { return "element block"; }
119
120 void finalize_parse()
121 {
122 if (!m_idsAssigned)
123 assign_ids();
124 }
125
126 private:
127 void handle_block_part(const std::string &name)
128 {
129 auto result = get_id_from_part_name(name, "BLOCK_");
130
131 if (!result.second)
132 return;
133
134 assign(name, result.first);
135 }
136
137 void assign_ids() const
138 {
139 unsigned nextPartId = 1;
140 for (const std::string &name : m_partNames) {
141 if (m_ids.find(name) == m_ids.end()) {
142 while (is_assigned(nextPartId))
143 nextPartId++;
144
145 assign(name, nextPartId);
146 }
147 }
148
149 m_idsAssigned = true;
150 }
151
152 void assign(const std::string &name, unsigned id) const
153 {
154 validate_name_and_id(name, id);
155 m_ids[name] = id;
156 m_parts[id] = name;
157 }
158
159 void validate_name_and_id(const std::string &name, unsigned id) const
160 {
161 if (is_registered(name)) {
162 if (m_ids[name] != id) {
163 std::ostringstream errmsg;
164 errmsg << "Cannot assign part '" << name << "' two different ids: " << m_ids[name]
165 << " and " << id;
166 m_errorHandler(errmsg);
167 }
168 }
169 else {
170 if (is_assigned(id)) {
171 std::ostringstream errmsg;
172 errmsg << "Part id " << id << " has already been assigned, cannot assign it to part '"
173 << name << "'";
174 m_errorHandler(errmsg);
175 }
176 }
177 }
178
179 bool is_assigned(unsigned id) const { return m_parts.count(id) > 0; }
180
181 unsigned get_part_id(const std::string &name) const
182 {
183 auto it = m_ids.find(name);
184 if (it == m_ids.end()) {
185 std::ostringstream errmsg;
186 errmsg << "PartIdMapping has no ID for invalid part name " << name;
187 m_errorHandler(errmsg);
188 }
189 return it->second;
190 }
191
192 std::string get_part_name(unsigned id) const
193 {
194 auto it = m_parts.find(id);
195 if (it == m_parts.end()) {
196 std::ostringstream errmsg;
197 errmsg << "PartIdMapping has no part name for invalid id " << id;
198 m_errorHandler(errmsg);
199 }
200 return it->second;
201 }
202
203 std::vector<std::string> m_partNames{};
204 mutable std::unordered_map<std::string, unsigned> m_ids;
205 mutable std::map<unsigned, std::string> m_parts;
206 mutable bool m_idsAssigned{false};
207
208 ErrorHandler m_errorHandler;
209 };

211 template <typename EntityId> class Coordinates
212 {
213 public:
214 Coordinates()
215 {
216 set_error_handler([](const std::ostringstream &errmsg) { default_error_handler(errmsg); });
217 }
218
219 const std::vector<double> &operator[](const EntityId nodeId) const
220 {
221 auto it(m_nodalCoords.find(nodeId));
222
223 if (it == m_nodalCoords.end()) {
224 std::ostringstream errmsg;
225 errmsg << "Could not find node id " << nodeId;
226 m_errorHandler(errmsg);
227 }
228
229 return it->second;
230 }
231
232 void set_coordinate_data(const unsigned spatialDim, const std::set<EntityId> &nodeIds,
233 const std::vector<double> &coordinates)
234 {
235 if (!coordinates.empty()) {
236 validate_num_coordinates(spatialDim, nodeIds, coordinates);
237 fill_coordinate_map(spatialDim, nodeIds, coordinates);
238 m_hasCoordinateData = true;
239 }
240 }
241
242 void set_error_handler(ErrorHandler errorHandler) { m_errorHandler = errorHandler; }
243
244 bool has_coordinate_data() const { return m_hasCoordinateData; }
245
246 private:
247 void validate_num_coordinates(const unsigned spatialDim, const std::set<EntityId> &nodeIds,
248 const std::vector<double> &coordinates)
249 {
250 if (coordinates.size() != nodeIds.size() * spatialDim) {
251 std::ostringstream errmsg;
252 errmsg << "Number of coordinates: " << coordinates.size()
253 << ", Number of nodes: " << nodeIds.size()
254 << ", Spatial dimension: " << spatialDim;
255 m_errorHandler(errmsg);
256 }
257 }
258
259 void fill_coordinate_map(const unsigned spatialDim, const std::set<EntityId> &nodeIds,
260 const std::vector<double> &coordinates)
261 {
262 std::vector<double>::const_iterator coordIter = coordinates.begin();
263 for (const EntityId &nodeId : nodeIds) {
264 m_nodalCoords[nodeId] = std::vector<double>(coordIter, coordIter + spatialDim);
265 coordIter += spatialDim;
266 }
267 }
268
269 bool m_hasCoordinateData{false};
270 std::unordered_map<EntityId, std::vector<double>> m_nodalCoords;
271 ErrorHandler m_errorHandler;
272 };

312 template <typename EntityId, typename Topology> struct TextMeshData
313 {
314 unsigned spatialDim{0};
315 std::vector<ElementData<EntityId, Topology>> elementDataVec{};
316 PartIdMapping partIds;
317 std::set<EntityId> nodeIds{};

319 Sidesets<EntityId, Topology> sidesets;
320 Nodesets<EntityId> nodesets;
321 Assemblies<EntityId> assemblies;
322
323 TextMeshData() : spatialDim(0) {}
324
325 void add_element(const ElementData<EntityId, Topology> &elem)
326 {
327 elementDataVec.push_back(elem);
328 for (const EntityId &nodeId : elem.nodeIds) {
329 nodeIds.insert(nodeId);
330 associate_node_with_proc(nodeId, elem.proc);
331 }
332 }
333
334 const std::set<EntityId> &nodes_on_proc(int proc) const
335 {
336 auto it = m_nodesOnProc.find(proc);
337 return it != m_nodesOnProc.end() ? it->second : m_emptyNodes;
338 }
339
340 unsigned num_nodes_on_proc(int proc) const
341 {
342 auto it = m_nodesOnProc.find(proc);
343 return it != m_nodesOnProc.end() ? it->second.size() : 0;
344 }
345
346 const std::set<int> &procs_for_node(const EntityId nodeId) const
347 {
348 auto it = m_procsForNode.find(nodeId);
349 return it != m_procsForNode.end() ? it->second : m_emptyProcs;
350 }
351
352 private:
353 void associate_node_with_proc(const EntityId nodeId, int proc)
354 {
355 m_procsForNode[nodeId].insert(proc);
356 m_nodesOnProc[proc].insert(nodeId);
357 }
358
359 std::unordered_map<EntityId, std::set<int>> m_procsForNode;
360 std::unordered_map<int, std::set<EntityId>> m_nodesOnProc;
361
362 std::set<int> m_emptyProcs{};
363 std::set<EntityId> m_emptyNodes{};
364 };
