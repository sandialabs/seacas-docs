66 // Return number of nodes in the connection shared with the donor zone.
67 size_t get_shared_node_count() const
68 {
69 size_t snc = 1;
70 for (int i = 0; i < 3; i++) {
71 snc *= (std::abs(m_ownerRangeEnd[i] - m_ownerRangeBeg[i]) + 1);
72 }
73 return snc;
74 }

110 // The following are all subsetted down to the portion that is actually on this zone
111 // This can be different than m_ownerRange and m_donorRange in a parallel run if the
112 // decomposition splits the connection. In a serial run, they are the same.
113 //
114 // 1 of ijk should be the same for rangeBeg and rangeEnd defining a surface.
115 Ioss::IJK_t m_ownerRangeBeg{}; // ijk triplet defining beginning of range on this zone
116 Ioss::IJK_t m_ownerRangeEnd{}; // ijk triplet defining end of range on this zone
117 Ioss::IJK_t m_ownerOffset{}; // ijk triplet with offset of this zone. Used to convert
118 // rangeBeg and rangeEnd global indices to local indices
119 Ioss::IJK_t m_donorRangeBeg{}; // ijk triplet defining beginning of range on the connected zone
120 Ioss::IJK_t m_donorRangeEnd{}; // ijk triplet defining end of range on the connected zone
121 Ioss::IJK_t m_donorOffset{}; // ijk triplet with offset of the donor zone. Used to convert
122 // donorRangeBeg and End global indices to local indices
123
124 size_t m_ownerGUID{}; // globally-unique id of owner
125 size_t m_donorGUID{}; // globally-unique id of donor
126
127 // NOTE: Shared nodes are "owned" by the zone with the lowest zone id.
128 int m_ownerZone{}; // "id" of zone that owns this connection
129 int m_donorZone{}; // "id" of zone that is donor (or other side) of this connection
130 int m_ownerProcessor{-1}; // processor that owns the owner zone
131 int m_donorProcessor{-1}; // processor that owns the donor zone
132 bool m_sameRange{false}; // True if owner and donor range should always match...(special use
133 // during decomp)
134 // True if it is the "lower" zone id in the connection. Uses adam unless both have same adam.
135 bool m_ownsSharedNodes{false}; // Deprecate soon
136
137 // True if this zc is created due to processor decompositions in a parallel run
138 mutable bool m_fromDecomp{false};
139
140 bool m_isActive{true}; // True if non-zero range. That is, it has at least one face
141
