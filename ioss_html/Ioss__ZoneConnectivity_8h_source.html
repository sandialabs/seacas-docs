68 // Return number of nodes in the connection shared with the donor zone.
69 size_t get_shared_node_count() const
70 {
71 size_t snc = 1;
72 for (int i = 0; i < 3; i++) {
73 snc *= (std::abs(m_ownerRangeEnd[i] - m_ownerRangeBeg[i]) + 1);
74 }
75 return snc;
76 }

112 // The following are all subsetted down to the portion that is actually on this zone
113 // This can be different than m_ownerRange and m_donorRange in a parallel run if the
114 // decomposition splits the connection. In a serial run, they are the same.
115 //
116 // 1 of ijk should be the same for rangeBeg and rangeEnd defining a surface.
117 Ioss::IJK_t m_ownerRangeBeg{}; // ijk triplet defining beginning of range on this zone
118 Ioss::IJK_t m_ownerRangeEnd{}; // ijk triplet defining end of range on this zone
119 Ioss::IJK_t m_ownerOffset{}; // ijk triplet with offset of this zone. Used to convert
120 // rangeBeg and rangeEnd global indices to local indices
121 Ioss::IJK_t m_donorRangeBeg{}; // ijk triplet defining beginning of range on the connected zone
122 Ioss::IJK_t m_donorRangeEnd{}; // ijk triplet defining end of range on the connected zone
123 Ioss::IJK_t m_donorOffset{}; // ijk triplet with offset of the donor zone. Used to convert
124 // donorRangeBeg and End global indices to local indices
125
126 size_t m_ownerGUID{}; // globally-unique id of owner
127 size_t m_donorGUID{}; // globally-unique id of donor
128
129 // NOTE: Shared nodes are "owned" by the zone with the lowest zone id.
130 int m_ownerZone{}; // "id" of zone that owns this connection
131 int m_donorZone{}; // "id" of zone that is donor (or other side) of this connection
132 int m_ownerProcessor{-1}; // processor that owns the owner zone
133 int m_donorProcessor{-1}; // processor that owns the donor zone
134 bool m_sameRange{false}; // True if owner and donor range should always match...(special use
135 // during decomp)
136 // True if it is the "lower" zone id in the connection. Uses adam unless both have same adam.
137 bool m_ownsSharedNodes{false}; // Deprecate soon
138
139 // True if this zc is created due to processor decompositions in a parallel run
140 mutable bool m_fromDecomp{false};
141
142 bool m_isActive{true}; // True if non-zero range. That is, it has at least one face
143
144 private:
145 bool equal_(const Ioss::ZoneConnectivity &rhs, bool quiet) const;
146 };
