30class IOSS_EXPORT ScopeGuardImplBase
31{
32 ScopeGuardImplBase &operator=(const ScopeGuardImplBase &) = delete;
33
34protected:
35 ~ScopeGuardImplBase() = default;
36 ScopeGuardImplBase(const ScopeGuardImplBase &other) : dismissed_(other.dismissed_)
37 {
38 other.Dismiss();
39 }
40
41 template <typename J> static void SafeExecute(J &j)
42 {
43 if (!j.dismissed_) {
44 try {
45 j.Execute();
46 }
47 catch (...) {
48 }
49 }
50 }
51
52 mutable bool dismissed_{false};
53
54public:
55 ScopeGuardImplBase() = default;
56 void Dismiss() const { dismissed_ = true; }
57};
