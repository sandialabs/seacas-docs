27class ScopeGuardImplBase
28{
29 ScopeGuardImplBase &operator=(const ScopeGuardImplBase &) = delete;
30
31protected:
32 ~ScopeGuardImplBase() = default;
33 ScopeGuardImplBase(const ScopeGuardImplBase &other) : dismissed_(other.dismissed_)
34 {
35 other.Dismiss();
36 }
37
38 template <typename J> static void SafeExecute(J &j)
39 {
40 if (!j.dismissed_) {
41 try {
42 j.Execute();
43 }
44 catch (...) {
45 }
46 }
47 }
48
49 mutable bool dismissed_{false};
50
51public:
52 ScopeGuardImplBase() = default;
53 void Dismiss() const { dismissed_ = true; }
54};
