88 {
89 const std::vector<double> coordinates{};

92

101
102 std::vector<SharedNode> sharedNodes{};
103
104 DashSurfaceData(std::vector<double> coords, std::vector<int> connectivity1,
105 std::vector<int> connectivity2)
106 : coordinates(std::move(coords)), surfaceAConnectivity(std::move(connectivity1)),
107 surfaceBConnectivity(std::move(connectivity2))
108 {
109 this->setSerialDefaults();
110 }
111
112 private:
113 void setSerialDefaults()
114 {
115 globalNumberOfNodes = coordinates.size() / SPATIAL_DIMENSION;
116
117 globalNumberOfElementsSurface1 = surfaceBConnectivity.size() / NUM_NODES_PER_QUAD_FACE;
118 globalNumberOfElementsSurface2 = surfaceAConnectivity.size() / NUM_NODES_PER_QUAD_FACE;
119 globalNumberOfElements = globalNumberOfElementsSurface1 + globalNumberOfElementsSurface2;
120
121 globalIdsOfLocalElements.resize(globalNumberOfElements);
122 globalIdsOfLocalNodes.resize(globalNumberOfNodes);
123
124 for (size_t i = 0; i < globalIdsOfLocalElements.size(); i++) {
125 globalIdsOfLocalElements[i] = i + 1;
126 }
127
128 for (size_t i = 0; i < globalIdsOfLocalNodes.size(); i++) {
129 globalIdsOfLocalNodes[i] = i + 1;
130 }
131 }
132 };
