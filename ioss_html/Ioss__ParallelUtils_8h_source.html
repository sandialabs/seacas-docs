137 template <typename T>
138 int MY_Alltoallv64(const std::vector<T> &sendbuf, const std::vector<int64_t> &sendcounts,
139 const std::vector<int64_t> &senddisp, std::vector<T> &recvbuf,
140 const std::vector<int64_t> &recvcounts, const std::vector<int64_t> &recvdisp,
141 MPI_Comm comm)
142 {
143 int processor_count = 0;
144 int my_processor = 0;
145 MPI_Comm_size(comm, &processor_count);
146 MPI_Comm_rank(comm, &my_processor);
147
148 // Verify that all 'counts' can fit in an integer. Symmetric
149 // communication, so recvcounts are sendcounts on another processor.
150 for (int i = 0; i < processor_count; i++) {
151 int snd_cnt = static_cast<int>(sendcounts[i]);
152 if (static_cast<int64_t>(snd_cnt) != sendcounts[i]) {
153 std::ostringstream errmsg;
154 errmsg << "ERROR: The number of items that must be communicated via MPI calls from\n"
155 << " processor " << my_processor << " to processor " << i << " is "
156 << sendcounts[i]
157 << "\n which exceeds the storage capacity of the integers "
158 "used by MPI functions.\n";
159 IOSS_ERROR(errmsg);
160 }
161 }
162
163 size_t pow_2 = Ioss::Utils::power_2(processor_count);
164
165 for (size_t i = 1; i < pow_2; i++) {
166 MPI_Status status{};
167
168 int tag = 24713;
169 size_t exchange_proc = i ^ my_processor;
170 if (exchange_proc < static_cast<size_t>(processor_count)) {
171 int snd_cnt = static_cast<int>(
172 sendcounts[exchange_proc]); // Converts from int64_t to int as needed by mpi
173 int rcv_cnt = static_cast<int>(recvcounts[exchange_proc]);
174 if (static_cast<size_t>(my_processor) < exchange_proc) {
175 MPI_Send((void *)&sendbuf[senddisp[exchange_proc]], snd_cnt, mpi_type(T(0)),
176 exchange_proc, tag, comm);
177 MPI_Recv(&recvbuf[recvdisp[exchange_proc]], rcv_cnt, mpi_type(T(0)), exchange_proc, tag,
178 comm, &status);
179 }
180 else {
181 MPI_Recv(&recvbuf[recvdisp[exchange_proc]], rcv_cnt, mpi_type(T(0)), exchange_proc, tag,
182 comm, &status);
183 MPI_Send((void *)&sendbuf[senddisp[exchange_proc]], snd_cnt, mpi_type(T(0)),
184 exchange_proc, tag, comm);
185 }
186 }
187 }
188
189 // Take care of this processor's data movement...
190 std::copy(&sendbuf[senddisp[my_processor]],
191 &sendbuf[senddisp[my_processor] + sendcounts[my_processor]],
192 &recvbuf[recvdisp[my_processor]]);
193 return 0;
194 }

196 template <typename T>
197 int MY_Alltoallv(const std::vector<T> &sendbuf, const std::vector<int64_t> &sendcnts,
198 const std::vector<int64_t> &senddisp, std::vector<T> &recvbuf,
199 const std::vector<int64_t> &recvcnts, const std::vector<int64_t> &recvdisp,
200 MPI_Comm comm)
201 {
202// Wrapper to handle case where send/recv counts and displacements are 64-bit integers.
203// Two cases:
204// 1) They are of type 64-bit integers, but only storing data in the 32-bit integer range.
205// -- if (sendcnts[#proc-1] + senddisp[#proc-1] < 2^31, then we are ok
206// 2) They are of type 64-bit integers, and storing data in the 64-bit integer range.
207// -- call special alltoallv which does point-to-point sends
208#if 1
209 int processor_count = 0;
210 MPI_Comm_size(comm, &processor_count);
211 size_t max_comm = sendcnts[processor_count - 1] + senddisp[processor_count - 1];
212 size_t one = 1;
213 if (max_comm < one << 31) {
214 // count and displacement data in range, need to copy to integer vector.
215 std::vector<int> send_cnt(sendcnts.begin(), sendcnts.end());
216 std::vector<int> send_dis(senddisp.begin(), senddisp.end());
217 std::vector<int> recv_cnt(recvcnts.begin(), recvcnts.end());
218 std::vector<int> recv_dis(recvdisp.begin(), recvdisp.end());
219 return MPI_Alltoallv((void *)sendbuf.data(), send_cnt.data(), send_dis.data(), mpi_type(T(0)),
220 (void *)recvbuf.data(), recv_cnt.data(), recv_dis.data(), mpi_type(T(0)),
221 comm);
222 }
223 else {
224#endif
225 // Same as if each processor sent a message to every other process with:
226 // MPI_Send(sendbuf+senddisp[i]*sizeof(sendtype),sendcnts[i], sendtype, i, tag, comm);
227 // And received a message from each processor with a call to:
228 // MPI_Recv(recvbuf+recvdisp[i]*sizeof(recvtype),recvcnts[i], recvtype, i, tag, comm);
229 return MY_Alltoallv64(sendbuf, sendcnts, senddisp, recvbuf, recvcnts, recvdisp, comm);
230#if 1
231 }
232#endif
233 }

248 template <typename T>
249 void ParallelUtils::global_array_minmax(std::vector<T> &local_minmax, MinMax which) const
250 {
251 PAR_UNUSED(local_minmax);
252 PAR_UNUSED(which);
253#ifdef SEACAS_HAVE_MPI
254 if (parallel_size() > 1 && !local_minmax.empty()) {
255 if (Ioss::SerializeIO::isEnabled() && Ioss::SerializeIO::inBarrier()) {
256 std::ostringstream errmsg;
257 errmsg << "Attempting mpi while in barrier owned by " << Ioss::SerializeIO::getOwner();
258 IOSS_ERROR(errmsg);
259 }
260
261 std::vector<T> maxout(local_minmax.size());
262 MPI_Op oper = MPI_MAX;
263 if (which == Ioss::ParallelUtils::DO_MAX) {
264 oper = MPI_MAX;
265 }
266 else if (which == Ioss::ParallelUtils::DO_MIN) {
267 oper = MPI_MIN;
268 }
269 else if (which == Ioss::ParallelUtils::DO_SUM) {
270 oper = MPI_SUM;
271 }
272
273 const int success =
274 MPI_Allreduce((void *)(local_minmax.data()), maxout.data(),
275 static_cast<int>(local_minmax.size()), mpi_type(T()), oper, communicator_);
276 if (success != MPI_SUCCESS) {
277 std::ostringstream errmsg;
278 errmsg << "Ioss::ParallelUtils::global_array_minmax - MPI_Allreduce failed";
279 IOSS_ERROR(errmsg);
280 }
281 // Now copy back into passed in array...
282 for (size_t i = 0; i < local_minmax.size(); i++) {
283 local_minmax[i] = maxout[i];
284 }
285 }
286#endif
287 }
