30 /*! The TopologyModified enumeration is used as an argument to the
31 * topology_modified() functions in io to
32 * specify the type of topology modification that has occurred. The
33 * calls to topology_modified() are cumulative between
34 * output steps, so a call with TOPOLOGY_REORDER followed by a call
35 * with TOPOLOGY_SHUFFLE will do the right thing. Typical examples
36 * of when these would be used are:
37 * - TOPOLOGY_SAME: No change, but easier to call function than not.
38 * - TOPOLOGY_REORDER: Element Death which reorders the Registrars
39 * - TOPOLOGY_SHUFFLE: Load Balancing
40 * - TOPOLOGY_HADAPT: H-Adaptivity
41 * - TOPOLOGY_GHOST: Ghost nodes/edges/faces/elements created/destroyed
42 * - TOPOLOGY_GEOMETRY: Model data is modified, overlap removal.
43 * - TOPOLOGY_CREATEDELETE: Surface erosion, particle creation
44 * - TOPOLOGY_UNKNOWN: Something else, catchall option.
45 */
46 enum TopologyModified {
47 TOPOLOGY_SAME = ( 0), //!< No change, also used for initialization
48 TOPOLOGY_REORDER = (1U << 0), //!< Data structures reordered on processor, no change between procs.
49 TOPOLOGY_SHUFFLE = (1U << 1), //!< Globally the same, data moved among processors.
50 TOPOLOGY_HADAPT = (1U << 2), //!< Elements split/combined; not moved cross-proc
51 TOPOLOGY_GEOMETRY = (1U << 3), //!< Geometry (mesh coordinates) modified. Restart needs to know this.
52 TOPOLOGY_CREATEFACE = (1U << 4), //!< Face/Edge are created/deleted.
53 TOPOLOGY_CREATEELEM = (1U << 5), //!< Elements are created/deleted.
54 TOPOLOGY_CREATENODE = (1U << 6), //!< Nodes are created/deleted.
55 TOPOLOGY_CREATEASSEMBLY = (1U << 7), //!< Assemblies are created/deleted.
56 TOPOLOGY_UNKNOWN = (1U << 8), //!< Unknown change, recreate from scratch.
57 };

161 ObserverType* typed_observer = dynamic_cast<ObserverType*>(observer.get());
162 if (typed_observer != nullptr) {
