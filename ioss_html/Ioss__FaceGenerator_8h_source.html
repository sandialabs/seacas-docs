80 struct IOSS_EXPORT FaceEqual
81 {
82 bool operator()(const Face &left, const Face &right) const
83 {
84 if (left.hashId_ != right.hashId_) {
85 return false;
86 }
87 // Hash (hashId_) is equal
88 // Check whether same vertices (can be in different order)
89 // Most (All?) of the time, there are no hashId_ collisions, so this test will not
90 // find a difference and the function will return 'true'
91 // However, for some reason, removing this check does not change the execution time
92 // appreiciably...
93 for (auto lvert : left.connectivity_) {
94 if (std::find(right.connectivity_.cbegin(), right.connectivity_.cend(), lvert) ==
95 right.connectivity_.cend()) {
96 // Not found, therefore not the same.
97 return false;
98 }
99 }
100 return true;
101 }
102 };
