83 template <typename INT>
84 static void map_cgns_connectivity(const Ioss::ElementTopology *topo, size_t element_count,
85 INT *idata)
86 {
87 // Map from CGNS to IOSS/Exodus/Patran order...
88 switch (topo->shape()) {
89 case Ioss::ElementShape::HEX:
90 switch (topo->number_nodes()) {
91 case 8:
92 case 20: break;
93 case 27: {
94 // nodes 1 through 20 are the same...
95 //
96 // ioss: 21, 22, 23, 24, 25, 26, 27 [zero-based: 20, 21, 22, 23, 24, 25, 26]
97 // cgns: 27, 21, 26, 25, 23, 22, 24 [zero-based: 26, 20, 25, 24, 22, 21, 23]
98 static std::array<int, 7> hex27_map{26, 20, 25, 24, 22, 21, 23};
99 for (size_t i = 0; i < element_count; i++) {
100 size_t con_beg = 27 * i; // start of connectivity for i'th element.
101 std::array<int, 7> reorder;
102 for (size_t j = 0; j < 7; j++) {
103 reorder[j] = idata[con_beg + hex27_map[j]];
104 }
105
106 for (size_t j = 0; j < 7; j++) {
107 idata[con_beg + 20 + j] = reorder[j];
108 }
109 }
110 }
111 }
112 break;
113 default:
114 // do nothing cgns ordering matches ioss/exodus/patran (or not handled yet... todo: add
115 // error checking)
116 ;
117 }
118 }

157 template <typename INT>
158 static void map_cgns_face_to_ioss(const Ioss::ElementTopology *parent_topo, size_t num_to_get,
159 INT *idata)
160 {
161 // The {topo}_map[] arrays map from CGNS face# to IOSS face#.
162 // See http://cgns.github.io/CGNS_docs_current/sids/conv.html#unstructgrid
163 // NOTE: '0' for first entry is to account for 1-based face numbering.
164
165 switch (parent_topo->shape()) {
166 case Ioss::ElementShape::HEX: {
167 static int hex_map[] = {0, 5, 1, 2, 3, 4, 6};
168 for (size_t i = 0; i < num_to_get; i++) {
169 idata[2 * i + 1] = hex_map[idata[2 * i + 1]];
170 }
171 } break;
172
173 case Ioss::ElementShape::TET: {
174 static int tet_map[] = {0, 4, 1, 2, 3};
175 for (size_t i = 0; i < num_to_get; i++) {
176 idata[2 * i + 1] = tet_map[idata[2 * i + 1]];
177 }
178 } break;
179
180 case Ioss::ElementShape::PYRAMID: {
181 static int pyr_map[] = {0, 5, 1, 2, 3, 4};
182 for (size_t i = 0; i < num_to_get; i++) {
183 idata[2 * i + 1] = pyr_map[idata[2 * i + 1]];
184 }
185 } break;
186
187 case Ioss::ElementShape::WEDGE:
188#if 0
189 static int wed_map[] = {0, 1, 2, 3, 4, 5}; // Same
190 // Not needed -- maps 1 to 1
191 for (size_t i=0; i < num_to_get; i++) {
192 idata[2*i+1] = wed_map[idata[2*i+1]];
193 }
194#endif
195 break;
196 default:;
197 }
198 }

200 static void map_ioss_face_to_cgns(const Ioss::ElementTopology *parent_topo, size_t num_to_get,
201 CGNSIntVector &data)
202 {
203 // The {topo}_map[] arrays map from CGNS face# to IOSS face#.
204 // See http://cgns.github.io/CGNS_docs_current/sids/conv.html#unstructgrid
205 // NOTE: '0' for first entry is to account for 1-based face numbering.
206
207 switch (parent_topo->shape()) {
208 case Ioss::ElementShape::HEX: {
209 static int hex_map[] = {0, 2, 3, 4, 5, 1, 6};
210 for (size_t i = 0; i < num_to_get; i++) {
211 data[num_to_get * 2 + i] = hex_map[data[num_to_get * 2 + i]];
212 }
213 } break;
214
215 case Ioss::ElementShape::TET: {
216 static int tet_map[] = {0, 2, 3, 4, 1};
217 for (size_t i = 0; i < num_to_get; i++) {
218 data[num_to_get * 2 + i] = tet_map[data[num_to_get * 2 + i]];
219 }
220 } break;
221
222 case Ioss::ElementShape::PYRAMID: {
223 static int pyr_map[] = {0, 2, 3, 4, 5, 1};
224 for (size_t i = 0; i < num_to_get; i++) {
225 data[num_to_get * 2 + i] = pyr_map[data[num_to_get * 2 + i]];
226 }
227 } break;
228
229 case Ioss::ElementShape::WEDGE:
230#if 0
231 static int wed_map[] = {0, 1, 2, 3, 4, 5}; // Same
232 // Not needed -- maps 1 to 1
233 for (size_t i=0; i < num_to_get; i++) {
234 data[num_to_get * 2 + i] = wed_map[data[num_to_get * 2 + i]];
235 }
236#endif
237 break;
238 default:;
239 }
