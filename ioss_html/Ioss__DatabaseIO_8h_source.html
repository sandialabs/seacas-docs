377 // The get_field and put_field functions are just a wrapper around the
378 // pure virtual get_field_internal and put_field_internal functions,
379 // but this lets me add some debug/checking/common code to the
380 // functions without having to do it in the calling code or in the
381 // derived classes code. This also fulfills the heuristic that a
382 // public interface should not contain pure virtual functions.
383 template <typename T>
384 int64_t get_field(const T *reg, const Field &field, void *data, size_t data_size) const
385 {
386 IOSS_FUNC_ENTER(m_);
387 verify_and_log(reg, field, 1);
388 int64_t retval = get_field_internal(reg, field, data, data_size);
389 if (get_nan_detection()) {
390 verify_field_data(reg, field, Ioss::Field::InOut::INPUT, data);
391 }
392 verify_and_log(nullptr, field, 1);
393 return retval;
394 }
395
396 template <typename T>
397 int64_t put_field(const T *reg, const Field &field, void *data, size_t data_size) const
398 {
399 IOSS_FUNC_ENTER(m_);
400 verify_and_log(reg, field, 0);
401 if (get_nan_detection()) {
402 verify_field_data(reg, field, Ioss::Field::InOut::OUTPUT, data);
403 }
404 int64_t retval = put_field_internal(reg, field, data, data_size);
405 verify_and_log(nullptr, field, 0);
406 return retval;
407 }

504 /*!
505 * The overlay_count specifies the number of restart outputs
506 * which will be overlaid on top of the currently written
507 * step before advancing to the next step on the restart
508 * database.
509 *
510 * For example, if restarts are being output every 0.1
511 * seconds and the overlay count is specified as 2, then
512 * restart will write time 0.1 to step 1 of the database.
513 * It will then write 0.2 and 0.3 also to step 1. It will
514 * then increment the database step and write 0.4 to step 2;
515 * overlay 0.5 and 0.6 on step 2... At the end of the
516 * analysis, assuming it runs to completion, the database
517 * would have times 0.3, 0.6, 0.9, ... However, if there
518 * were a problem during the analysis, the last step on the
519 * database would contain an intermediate step.
520 *
521 * The cycle_count specifies the number of restart steps
522 * which will be written to the restart database before
523 * previously written steps are overwritten.
524 *
525 * For example, if the cycle count is 5 and restart is written every 0.1
526 * seconds, the restart system will write data at times 0.1, 0.2, 0.3,
527 * 0.4, 0.5 to the database.
528 *
529 * It will then overwrite the first step with data from time 0.6, the
530 * second with time 0.7. At time 0.8, the database would contain data at
531 * times 0.6, 0.7, 0.8, 0.4, 0.5. Note that time will not necessarily be
532 * monotonically increasing on a database that specifies the cycle
533 * count.
534 *
535 * The cycle count and overlay count can both be used at the same time
536 * also. The basic formula is:
537 *
538 * db_step = (((output_step - 1) / overlay) % cycle) + 1
539 *
540 * where "output_step" is the step that this would have been on the
541 * database in a normal write (1,2,3,....) and "db_step" is the step
542 * number that this will be written to.
543 *
544 * If you only want the last step available on the database,
545 * use set_cycle_count(1)
546 */
547 void set_cycle_count(int count) const { cycleCount = count; }
548 int get_cycle_count() const { return cycleCount; }
549 void set_overlay_count(int count) const { overlayCount = count; }
550 int get_overlay_count() const { return overlayCount; }
551 void set_file_per_state(bool yes_no) const { filePerState = yes_no; }
552 bool get_file_per_state() const { return filePerState; }
