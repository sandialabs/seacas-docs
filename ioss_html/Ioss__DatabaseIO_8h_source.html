404 // The get_field and put_field functions are just a wrapper around the
405 // pure virtual get_field_internal and put_field_internal functions,
406 // but this lets me add some debug/checking/common code to the
407 // functions without having to do it in the calling code or in the
408 // derived classes code. This also fulfills the heuristic that a
409 // public interface should not contain pure virtual functions.
410 template <typename T>
411 int64_t get_field(const T *reg, const Field &field, void *data, size_t data_size) const
412 {
413 IOSS_FUNC_ENTER(m_);
414 verify_and_log(reg, field, 1);
415 int64_t retval = get_field_internal(reg, field, data, data_size);
416 if (get_nan_detection()) {
417 verify_field_data(reg, field, Ioss::Field::InOut::INPUT, data);
418 }
419 verify_and_log(nullptr, field, 1);
420 return retval;
421 }
422
423 template <typename T>
424 int64_t put_field(const T *reg, const Field &field, void *data, size_t data_size) const
425 {
426 IOSS_FUNC_ENTER(m_);
427 verify_and_log(reg, field, 0);
428 if (get_nan_detection()) {
429 verify_field_data(reg, field, Ioss::Field::InOut::OUTPUT, data);
430 }
431 int64_t retval = put_field_internal(reg, field, data, data_size);
432 verify_and_log(nullptr, field, 0);
433 return retval;
434 }

536 /*!
537 * The overlay_count specifies the number of restart outputs
538 * which will be overlaid on top of the currently written
539 * step before advancing to the next step on the restart
540 * database.
541 *
542 * For example, if restarts are being output every 0.1
543 * seconds and the overlay count is specified as 2, then
544 * restart will write time 0.1 to step 1 of the database.
545 * It will then write 0.2 and 0.3 also to step 1. It will
546 * then increment the database step and write 0.4 to step 2;
547 * overlay 0.5 and 0.6 on step 2... At the end of the
548 * analysis, assuming it runs to completion, the database
549 * would have times 0.3, 0.6, 0.9, ... However, if there
550 * were a problem during the analysis, the last step on the
551 * database would contain an intermediate step.
552 *
553 * The cycle_count specifies the number of restart steps
554 * which will be written to the restart database before
555 * previously written steps are overwritten.
556 *
557 * For example, if the cycle count is 5 and restart is written every 0.1
558 * seconds, the restart system will write data at times 0.1, 0.2, 0.3,
559 * 0.4, 0.5 to the database.
560 *
561 * It will then overwrite the first step with data from time 0.6, the
562 * second with time 0.7. At time 0.8, the database would contain data at
563 * times 0.6, 0.7, 0.8, 0.4, 0.5. Note that time will not necessarily be
564 * monotonically increasing on a database that specifies the cycle
565 * count.
566 *
567 * The cycle count and overlay count can both be used at the same time
568 * also. The basic formula is:
569 *
570 * db_step = (((output_step - 1) / overlay) % cycle) + 1
571 *
572 * where "output_step" is the step that this would have been on the
573 * database in a normal write (1,2,3,....) and "db_step" is the step
574 * number that this will be written to.
575 *
576 * If you only want the last step available on the database,
577 * use set_cycle_count(1)
578 */
579 void set_cycle_count(int count) const { cycleCount = count; }
580 IOSS_NODISCARD int get_cycle_count() const { return cycleCount; }
581 void set_overlay_count(int count) const { overlayCount = count; }
582 IOSS_NODISCARD int get_overlay_count() const { return overlayCount; }
