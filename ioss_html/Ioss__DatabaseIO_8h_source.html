375 // The get_field and put_field functions are just a wrapper around the
376 // pure virtual get_field_internal and put_field_internal functions,
377 // but this lets me add some debug/checking/common code to the
378 // functions without having to do it in the calling code or in the
379 // derived classes code. This also fulfills the heuristic that a
380 // public interface should not contain pure virtual functions.
381 template <typename T>
382 int64_t get_field(const T *reg, const Field &field, void *data, size_t data_size) const
383 {
384 IOSS_FUNC_ENTER(m_);
385 verify_and_log(reg, field, 1);
386 int64_t retval = get_field_internal(reg, field, data, data_size);
387 verify_and_log(nullptr, field, 1);
388 return retval;
389 }
390
391 template <typename T>
392 int64_t put_field(const T *reg, const Field &field, void *data, size_t data_size) const
393 {
394 IOSS_FUNC_ENTER(m_);
395 verify_and_log(reg, field, 0);
396 int64_t retval = put_field_internal(reg, field, data, data_size);
397 verify_and_log(nullptr, field, 0);
398 return retval;
399 }

483 /*!
484 * The overlay_count specifies the number of restart outputs
485 * which will be overlaid on top of the currently written
486 * step before advancing to the next step on the restart
487 * database.
488 *
489 * For example, if restarts are being output every 0.1
490 * seconds and the overlay count is specified as 2, then
491 * restart will write time 0.1 to step 1 of the database.
492 * It will then write 0.2 and 0.3 also to step 1. It will
493 * then increment the database step and write 0.4 to step 2;
494 * overlay 0.5 and 0.6 on step 2... At the end of the
495 * analysis, assuming it runs to completion, the database
496 * would have times 0.3, 0.6, 0.9, ... However, if there
497 * were a problem during the analysis, the last step on the
498 * database would contain an intermediate step.
499 *
500 * The cycle_count specifies the number of restart steps
501 * which will be written to the restart database before
502 * previously written steps are overwritten.
503 *
504 * For example, if the cycle count is 5 and restart is written every 0.1
505 * seconds, the restart system will write data at times 0.1, 0.2, 0.3,
506 * 0.4, 0.5 to the database.
507 *
508 * It will then overwrite the first step with data from time 0.6, the
509 * second with time 0.7. At time 0.8, the database would contain data at
510 * times 0.6, 0.7, 0.8, 0.4, 0.5. Note that time will not necessarily be
511 * monotonically increasing on a database that specifies the cycle
512 * count.
513 *
514 * The cycle count and overlay count can both be used at the same time
515 * also. The basic formula is:
516 *
517 * db_step = (((output_step - 1) / overlay) % cycle) + 1
518 *
519 * where "output_step" is the step that this would have been on the
520 * database in a normal write (1,2,3,....) and "db_step" is the step
521 * number that this will be written to.
522 *
523 * If you only want the last step available on the database,
524 * use set_cycle_count(1)
525 */
526 void set_cycle_count(int count) const { cycleCount = count; }
527 int get_cycle_count() const { return cycleCount; }
528 void set_overlay_count(int count) const { overlayCount = count; }
529 int get_overlay_count() const { return overlayCount; }
530 void set_file_per_state(bool yes_no) const { filePerState = yes_no; }
531 bool get_file_per_state() const { return filePerState; }
