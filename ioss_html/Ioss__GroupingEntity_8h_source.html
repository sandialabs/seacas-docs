31 /** \brief Base class for all 'grouping' entities.
32 * The following derived classes are typical:
33 *
34 * -- NodeSet -- grouping of nodes (0d topology)
35 *
36 * -- EdgeSet -- grouping of edges (1d topology)
37 *
38 * -- FaceSet -- grouping of faces (2d topology) [Surface]
39 *
40 * Similarly, there is:
41 *
42 * -- NodeBlock -- grouping of 'similar' nodes (same degree of freedom, ...)
43 *
44 * -- ElementBlock -- grouping of 'similar' elements (same element topology,
45 * attributes, ...)
46 * 0d, 1d, 2d, 3d topology possible -- e.g., sphere, bar, quad, hex
47 *
48 * A Region is also a grouping entity, except that its list of subentites
49 * are other GroupingEntities. That is, it maintains a list of NodeBlocks,
50 * ElementBlocks, NodeLists, CommLists and Surfaces. [Similar to the
51 * "Composite Patter" in Design Patterns] All interface to GroupingEntities
52 * is through the Region class; clients of the IO subsystem have no direct
53 * access to the underlying GroupingEntities (other than the Region).
54 *
55 * Each GroupingEntity contains:
56 *
57 * -- name
58 *
59 * -- MeshEntities of the specified topological dimension
60 *
61 * -- Optional attributes, either global (applied to the groupingentity), or
62 * unique value(s) to be applied to each subentity.
63 *
64 * -- Data items
65 */

502/** \brief Read type 'T' field data from the database file into memory using a std::vector.
503 *
504 * \param[in] field_name The name of the field to read.
505 * \param[out] data The data.
506 * \returns The number of values read.
507 *
508 */
509template <typename T>
510int64_t Ioss::GroupingEntity::get_field_data(const std::string &field_name,
511 std::vector<T> &data) const
512{
513 verify_field_exists(field_name, "input");
514
515 Ioss::Field field = get_field(field_name);
516 field.check_type(Ioss::Field::get_field_type(T(0)));
517
518 data.resize(field.raw_count() * field.raw_storage()->component_count());
519 size_t data_size = data.size() * sizeof(T);
520 auto retval = internal_get_field_data(field, data.data(), data_size);
521
522 // At this point, transform the field if specified...
523 if (retval >= 0) {
524 field.transform(data.data());
525 }
526
527 return retval;
528}
529
530/** \brief Write type 'T' field data from memory into the database file using a std::vector.
531 *
532 * \param[in] field_name The name of the field to write.
533 * \param[in] data The data.
534 * \returns The number of values written.
535 *
536 */
537template <typename T>
538int64_t Ioss::GroupingEntity::put_field_data(const std::string &field_name,
539 const std::vector<T> &data) const
540{
541 verify_field_exists(field_name, "output");
542
543 Ioss::Field field = get_field(field_name);
544 field.check_type(Ioss::Field::get_field_type(T(0)));
545 size_t data_size = data.size() * sizeof(T);
546 if (field.has_transform()) {
547 // Need non-const data since the transform will change the users data.
548 std::vector<T> nc_data(data);
549 field.transform(nc_data.data());
550 return internal_put_field_data(field, nc_data.data(), data_size);
551 }
552
553 T *my_data = const_cast<T *>(data.data());
554 return internal_put_field_data(field, my_data, data_size);
555}
556
557template <typename T>
558int64_t Ioss::GroupingEntity::put_field_data(const std::string &field_name,
559 std::vector<T> &data) const
560{
561 verify_field_exists(field_name, "output");
562
563 Ioss::Field field = get_field(field_name);
564 field.check_type(Ioss::Field::get_field_type(T(0)));
565 size_t data_size = data.size() * sizeof(T);
566 T *my_data = const_cast<T *>(data.data());
567 field.transform(my_data);
568 return internal_put_field_data(field, my_data, data_size);
569}

573/** \brief Read field data from the database file into memory using a 1-D Kokkos:::View.
574 *
575 * \tparam T The data type.
576 * \tparam Args The other template arguments for data.
577 * \param[in] field_name The name of the field to read.
578 * \param[out] data The data.
579 * \returns The number of values read.
580 *
581 */
582template <typename T, typename... Args>
583int64_t Ioss::GroupingEntity::get_field_data(const std::string &field_name,
584 Kokkos::View<T *, Args...> &data) const
585{
586 typedef Kokkos::View<T *, Args...> ViewType;
587
588 verify_field_exists(field_name, "input");
589
590 Ioss::Field field = get_field(field_name);
591
592 // Resize the view
593 auto new_view_size = field.raw_count() * field.raw_storage()->component_count();
594 Kokkos::resize(data, new_view_size);
595 size_t data_size = new_view_size * sizeof(T);
596
597 // Create a host mirror view. (No memory allocation if data is in HostSpace.)
598 typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
599
600 // Extract a pointer to the underlying allocated memory of the host view.
601 T *host_data_ptr = host_data.data();
602
603 // Extract the data from disk to the underlying memory pointed to by host_data_ptr.
604 auto retval = internal_get_field_data(field, host_data_ptr, data_size);
605
606 // At this point, transform the field if specified...
607 if (retval >= 0)
608 field.transform(host_data_ptr);
609
610 // Copy the data to the device. (No op if data is in HostSpace.)
611 Kokkos::deep_copy(data, host_data);
612
613 return retval;
614}

616/** \brief Read field data from the database file into memory using a 2-D Kokkos:::View.
617 *
618 * \tparam T The data type
619 * \tparam Args The other template arguments for data.
620 * \param[in] field_name The name of the field to read.
621 * \param[out] data The data.
622 * \returns The number of values read.
623 *
624 */
625template <typename T, typename... Args>
626int64_t Ioss::GroupingEntity::get_field_data(const std::string &field_name,
627 Kokkos::View<T **, Args...> &data) const
628{
629 typedef Kokkos::View<T **, Args...> ViewType;
630
631 verify_field_exists(field_name, "input");
632
633 Ioss::Field field = get_field(field_name);
634
635 // Resize the view
636 int new_view_size_left = field.raw_count();
637 int new_view_size_right = field.raw_storage()->component_count();
638 Kokkos::resize(data, new_view_size_left, new_view_size_right);
639 size_t data_size = new_view_size_left * new_view_size_right * sizeof(T);
640
641 // Create and allocate an array to hold the data temporarily.
642 // This is necessary to ensure the data is placed in the correct
643 // location in the 2-D array, avoiding incorrect placement due
644 // to Views with padded dimensions.
645 T *data_array = new T[data_size];
646
647 // Create a host mirror view. (No memory allocation if data is in HostSpace.)
648 typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
649
650 // Extract the data from disk to the underlying memory pointed to by host_data_ptr.
651 auto retval = internal_get_field_data(field, data_array, data_size);
652
653 // At this point, transform the field if specified...
654 if (retval >= 0)
655 field.transform(data_array);
656
657 // Copy the data to the host Mirror view.
658 // The host mirror view has the same layout as the device view.
659 // For CUDA, this will be LayoutLeft. In this case, the loop order
660 // chosen here will be slower than the reversed loop order.
661 // However, The time for this extra in-memory copy is small
662 // compared with the time to copy from disk into memory.
663 for (int i = 0; i < new_view_size_left; ++i) {
664 for (int j = 0; j < new_view_size_right; ++j) {
665 host_data(i, j) = data_array[new_view_size_right * i + j];
666 }
667 }
668
669 // Delete the temporary array
670 delete[] data_array;
671
672 // Copy the data to the device. (No op if data is in HostSpace.)
673 Kokkos::deep_copy(data, host_data);
674
675 return retval;
676}

678/** \brief Write field data from memory into the database file using a 1-D Kokkos::View.
679 *
680 * \tparam T The data type
681 * \tparam Args The other template arguments for data.
682 * \param[in] field_name The name of the field to write.
683 * \param[in] data The data.
684 * \returns The number of values written.
685 *
686 */
687template <typename T, typename... Args>
688int64_t Ioss::GroupingEntity::put_field_data(const std::string &field_name,
689 Kokkos::View<T *, Args...> &data) const
690{
691 typedef Kokkos::View<T *, Args...> ViewType;
692
693 verify_field_exists(field_name, "output");
694
695 Ioss::Field field = get_field(field_name);
696 size_t data_size = field.raw_count() * field.raw_storage()->component_count() * sizeof(T);
697
698 // Create a host mirror view. (No memory allocation if data is in HostSpace.)
699 typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
700
701 // Copy the data to the host. (No op if data is in HostSpace.)
702 Kokkos::deep_copy(host_data, data);
703
704 // Extract a pointer to the underlying allocated memory of the host view.
705 T *host_data_ptr = host_data.data();
706
707 // Transform the field
708 field.transform(host_data_ptr);
709
710 // Copy the data to disk from the underlying memory pointed to by host_data_ptr.
711 return internal_put_field_data(field, host_data_ptr, data_size);
712}

714/** \brief Write field data from memory into the database file using a 2-D Kokkos::View.
715 *
716 * \tparam T The data type
717 * \tparam Args The other template arguments for data.
718 * \param[in] field_name The name of the field to write.
719 * \param[in] data The data.
720 * \returns The number of values written.
721 *
722 */
723template <typename T, typename... Args>
724int64_t Ioss::GroupingEntity::put_field_data(const std::string &field_name,
725 Kokkos::View<T **, Args...> &data) const
726{
727 typedef Kokkos::View<T **, Args...> ViewType;
728
729 verify_field_exists(field_name, "output");
730
731 Ioss::Field field = get_field(field_name);
732
733 int view_size_left = data.extent(0);
734 int view_size_right = data.extent(1);
735 size_t data_size = field.raw_count() * field.raw_storage()->component_count() * sizeof(T);
736
737 if (view_size_left * view_size_right * sizeof(T) != data_size) {
738 std::ostringstream errmsg;
739 errmsg << "\nERROR: View dimensions are inconsistent with field raw count or raw storage "
740 "component count"
741 << "for field" << field_name << "\n\n";
742 IOSS_ERROR(errmsg);
743 }
744
745 // Create a host mirror view. (No memory allocation if data is in HostSpace.)
746 typename ViewType::HostMirror host_data = Kokkos::create_mirror_view(data);
747
748 // Copy the data to the host. (No op if data is in HostSpace.)
749 Kokkos::deep_copy(host_data, data);
750
751 // Create and allocate an array to hold the data temporarily.
752 // This is necessary to ensure the data is taken from the correct
753 // location in the 2-D array, avoiding incorrect location due
754 // to Views with padded dimensions.
755 T *data_array = new T[data_size];
756
757 // Copy the data from the host Mirror view.
758 // The host mirror view has the same layout as the device view.
759 // For CUDA, this will be LayoutLeft. In this case, the loop order
760 // chosen here will be slower than the reversed loop order.
761 // However, The time for this extra in-memory copy is small
762 // compared with the time to copy to disk from memory.
763 for (int i = 0; i < view_size_left; ++i) {
764 for (int j = 0; j < view_size_right; ++j) {
765 data_array[view_size_right * i + j] = host_data(i, j);
766 }
767 }
768
769 // Transform the field
770 field.transform(data_array);
771
772 // Copy the data to disk from the underlying memory pointed to by data_array.
773 auto retval = internal_put_field_data(field, data_array, data_size);
774
775 // Delete the temporary array
776 delete[] data_array;
777
778 return retval;
779}
