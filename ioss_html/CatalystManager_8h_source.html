112 bool canDoOperation(unsigned int &operationCount)
113 {
114 bool retVal = true;
115 operationCount++;
116 if (operationCount == getNumberOfInputs()) {
117 operationCount = 0;
118 }
119 else {
120 retVal = false;
121 }
122 return retVal;
123 }

125 unsigned int getNumberOfInputs()
126 {
127 unsigned int retVal = 0;
128 if (getDataDescription() != nullptr) {
129 retVal = getDataDescription()->GetNumberOfInputDescriptions();
130 }
131 return retVal;
