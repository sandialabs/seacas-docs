199 // T/F if node with global index node owned by this processors ioss-decomp.
200 bool i_own_node(size_t global_index) const
201 {
202 // global_index is 1-based index into global list of nodes [1..global_node_count]
203 return std::binary_search(nodeGTL.begin(), nodeGTL.end(), global_index);

205
206 // T/F if element with global index elem owned by this processors ioss-decomp.
207 bool i_own_elem(size_t global_index) const
208 {
209 // global_index is 1-based index into global list of elements [1..global_element_count]
210#if defined(DC_USE_VECTOR)
211 return std::binary_search(
212 elemGTL.begin(), elemGTL.end(), std::pair<INT, INT>{global_index, 0},
213 [](const std::pair<INT, INT> &lhs, const std::pair<INT, INT> &val) -> bool {
214 return lhs.first < val.first;
215 });
216#else
217 return elemGTL.find(global_index) != elemGTL.end();
218#endif
219 }
220
221 size_t node_global_to_local(size_t global_index) const
222 {
223 // global_index is 1-based index into global list of nodes [1..global_node_count]
224 // return value is 1-based index into local list of nodes on this
225 // processor (ioss-decomposition)
226 // Note that for 'int', equivalence and equality are the same, so
227 // lower_bound is OK here (EffectiveSTL, Item 19)
228 typename std::vector<INT>::const_iterator I =
229 lower_bound(nodeGTL.begin(), nodeGTL.end(), global_index);
230 assert(I != nodeGTL.end());
231 return std::distance(nodeGTL.begin(), I) + 1; // Convert to 1-based index.
232 }
233
234 size_t elem_global_to_local(size_t global_index) const
235 {
236 // global_index is 1-based index into global list of elements [1..global_node_count]
237 // return value is 1-based index into local list of elements on this
238 // processor (ioss-decomposition)
239#if defined(DC_USE_VECTOR)
240 auto I = lower_bound(
241 elemGTL.begin(), elemGTL.end(), global_index,
242 [](const std::pair<INT, INT> &lhs, INT val) -> bool { return lhs.first < val; });
243 assert(I != elemGTL.end() && I->first == global_index);
244#else
245 auto I = elemGTL.find(global_index);
246#endif
247 assert(I != elemGTL.end());
248 assert(I->first == global_index);
249 return I->second;
250 }

315 // The following function is used if reading all element data on a
316 // processor instead of just an element blocks worth...
317 template <typename T>
318 void communicate_element_data(T *file_data, T *ioss_data, size_t comp_count) const
319 {
320 show_progress(__func__);
321 if (m_method == "LINEAR") {
322 assert(m_importPreLocalElemIndex == 0);
323 assert(exportElementMap.size() == 0);
324 assert(importElementMap.size() == 0);
325 // For "LINEAR" decomposition method, the `file_data` is the
326 // same as `ioss_data` Transfer all local data from file_data
327 // to ioss_data...
328 auto size = localElementMap.size() * comp_count;
329 std::copy(file_data, file_data + size, ioss_data);
330 return;
331 }
332
333 // Transfer the file-decomposition based data in 'file_data' to
334 // the ioss-decomposition based data in 'ioss_data'
335 std::vector<T> export_data(exportElementMap.size() * comp_count);
336 std::vector<T> import_data(importElementMap.size() * comp_count);
337
338 if (comp_count == 1) {
339 for (size_t i = 0; i < exportElementMap.size(); i++) {
340 size_t index = exportElementMap[i] - m_elementOffset;
341 export_data[i] = file_data[index];
342 }
343
344 // Transfer all local data from file_data to ioss_data...
345 for (size_t i = 0; i < localElementMap.size(); i++) {
346 size_t index = localElementMap[i];
347 ioss_data[m_importPreLocalElemIndex + i] = file_data[index];
348 }
349
350 // Get my imported data and send my exported data...
351 Ioss::MY_Alltoallv(export_data, exportElementCount, exportElementIndex, import_data,
352 importElementCount, importElementIndex, m_comm);
353 show_progress("\tCommunication 1a finished");
354
355 // Copy the imported data into ioss_data...
356 // Some comes before the local data...
357 for (size_t i = 0; i < m_importPreLocalElemIndex; i++) {
358 ioss_data[i] = import_data[i];
359 }
360
361 // Some comes after the local data...
362 size_t offset = m_importPreLocalElemIndex + localElementMap.size();
363 for (size_t i = 0; i < importElementMap.size() - m_importPreLocalElemIndex; i++) {
364 ioss_data[offset + i] = import_data[m_importPreLocalElemIndex + i];
365 }
366 }
367 else {
368 for (size_t i = 0; i < exportElementMap.size(); i++) {
369 size_t index = exportElementMap[i] - m_elementOffset;
370 for (size_t j = 0; j < comp_count; j++) {
371 export_data[comp_count * i + j] = file_data[comp_count * index + j];
372 }
373 }
374
375 // Transfer all local data from file_data to ioss_data...
376 for (size_t i = 0; i < localElementMap.size(); i++) {
377 size_t index = localElementMap[i];
378 for (size_t j = 0; j < comp_count; j++) {
379 ioss_data[comp_count * (m_importPreLocalElemIndex + i) + j] =
380 file_data[comp_count * index + j];
381 }
382 }
383
384 std::vector<INT> export_count(exportElementCount.begin(), exportElementCount.end());
385 std::vector<INT> export_disp(exportElementIndex.begin(), exportElementIndex.end());
386 std::vector<INT> import_count(importElementCount.begin(), importElementCount.end());
387 std::vector<INT> import_disp(importElementIndex.begin(), importElementIndex.end());
388
389 for (int i = 0; i < m_processorCount; i++) {
390 export_count[i] *= comp_count;
391 export_disp[i] *= comp_count;
392 import_count[i] *= comp_count;
393 import_disp[i] *= comp_count;
394 }
395
396 // Get my imported data and send my exported data...
397 Ioss::MY_Alltoallv(export_data, export_count, export_disp, import_data, import_count,
398 import_disp, m_comm);
399 show_progress("\tCommunication 1b finished");
400
401 // Copy the imported data into ioss_data...
402 // Some comes before the local data...
403 for (size_t i = 0; i < m_importPreLocalElemIndex; i++) {
404 for (size_t j = 0; j < comp_count; j++) {
405 ioss_data[comp_count * i + j] = import_data[comp_count * i + j];
406 }
407 }
408
409 // Some comes after the local data...
410 size_t offset = m_importPreLocalElemIndex + localElementMap.size();
411 for (size_t i = 0; i < importElementMap.size() - m_importPreLocalElemIndex; i++) {
412 for (size_t j = 0; j < comp_count; j++) {
413 ioss_data[comp_count * (offset + i) + j] =
414 import_data[comp_count * (m_importPreLocalElemIndex + i) + j];
415 }
416 }
417 }

421 void communicate_set_data(T *file_data, T *ioss_data, const SetDecompositionData &set,
422 size_t comp_count) const
423 {
424 show_progress(__func__);
425 std::vector<T> recv_data;
426
427 size_t size = set.file_count() * comp_count;
428 if (size == 0) {
429 return;
430 }
431
432 if (set.setComm_ != Ioss::ParallelUtils::comm_null()) {
433 recv_data.resize(size);
434 if (m_processor == set.root_) {
435 std::copy(file_data, file_data + size, recv_data.begin());
436 }
437 // NOTE: This broadcast uses a split communicator, so possibly
438 // not all processors participating.
439 Ioss::ParallelUtils pu(set.setComm_);
440 pu.broadcast(recv_data);
441 }
442 if (comp_count == 1) {
443 if (set.root_ == m_processor) {
444 for (size_t i = 0; i < set.ioss_count(); i++) {
445 size_t index = set.entitylist_map[i];
446 ioss_data[i] = file_data[index];
447 }
448 }
449 else {
450 // Receiving data from root...
451 for (size_t i = 0; i < set.ioss_count(); i++) {
452 size_t index = set.entitylist_map[i];
453 ioss_data[i] = recv_data[index];
454 }
455 }
456 }
457 else {
458 if (set.root_ == m_processor) {
459 for (size_t i = 0; i < set.ioss_count(); i++) {
460 size_t index = set.entitylist_map[i];
461 for (size_t j = 0; j < comp_count; j++) {
462 ioss_data[comp_count * i + j] = file_data[comp_count * index + j];
463 }
464 }
465 }
466 else {
467 // Receiving data from root...
468 for (size_t i = 0; i < set.ioss_count(); i++) {
469 size_t index = set.entitylist_map[i];
470 for (size_t j = 0; j < comp_count; j++) {
471 ioss_data[comp_count * i + j] = recv_data[comp_count * index + j];

473 }
474 }

478 template <typename T, typename U>
479 void communicate_block_data(T *file_data, U *ioss_data, const BlockDecompositionData &block,
480 size_t comp_count) const
481 {
482 show_progress(__func__);
483 if (m_method == "LINEAR") {
484 assert(block.localIossOffset == 0);
485 assert(block.exportMap.empty());
486 assert(block.importMap.empty());
487 // For "LINEAR" decomposition method, the `file_data` is the
488 // same as `ioss_data` Transfer all local data from file_data
489 // to ioss_data...
490 auto size = block.localMap.size() * comp_count;
491 std::copy(file_data, file_data + size, ioss_data);
492 return;
493 }
494
495 std::vector<U> exports;
496 exports.reserve(comp_count * block.exportMap.size());
497 std::vector<U> imports(comp_count * block.importMap.size());
498
499 if (comp_count == 1) {
500 for (int i : block.exportMap) {
501 exports.push_back(file_data[i]);
502 }
503
504 // Get my imported data and send my exported data...
505 Ioss::MY_Alltoallv(exports, block.exportCount, block.exportIndex, imports,
506 block.importCount, block.importIndex, m_comm);
507
508 // Map local and imported data to ioss_data.
509 for (size_t i = 0; i < block.localMap.size(); i++) {
510 ioss_data[i + block.localIossOffset] = file_data[block.localMap[i]];
511 }
512
513 for (size_t i = 0; i < block.importMap.size(); i++) {
514 ioss_data[block.importMap[i]] = imports[i];
515 }
516 }
517 else {
518 for (int i : block.exportMap) {
519 for (size_t j = 0; j < comp_count; j++) {
520 exports.push_back(file_data[i * comp_count + j]);
521 }
522 }
523
524 std::vector<int> export_count(block.exportCount.begin(), block.exportCount.end());
525 std::vector<int> export_disp(block.exportIndex.begin(), block.exportIndex.end());
526 std::vector<int> import_count(block.importCount.begin(), block.importCount.end());
527 std::vector<int> import_disp(block.importIndex.begin(), block.importIndex.end());
528
529 for (int i = 0; i < m_processorCount; i++) {
530 export_count[i] *= comp_count;
531 export_disp[i] *= comp_count;
532 import_count[i] *= comp_count;
533 import_disp[i] *= comp_count;
534 }
535
536 // Get my imported data and send my exported data...
537 Ioss::MY_Alltoallv(exports, export_count, export_disp, imports, import_count, import_disp,
538 m_comm);
539 show_progress("\tCommunication 1 finished");
540
541 // Map local and imported data to ioss_data.
542 for (size_t i = 0; i < block.localMap.size(); i++) {
543 for (size_t j = 0; j < comp_count; j++) {
544 ioss_data[(i + block.localIossOffset) * comp_count + j] =
545 file_data[block.localMap[i] * comp_count + j];
546 }
547 }
548
549 for (size_t i = 0; i < block.importMap.size(); i++) {
550 for (size_t j = 0; j < comp_count; j++) {
551 ioss_data[block.importMap[i] * comp_count + j] = imports[i * comp_count + j];
552 }
553 }
554 }

557 template <typename T, typename U>
558 std::vector<size_t> do_communicate_entity_data(
559 T *file_data, U *ioss_data, const std::vector<BlockDecompositionData> &blocks,
560 const std::vector<int64_t> &blockSubsetIndex, const std::vector<size_t> &fileOffset,
561 const std::vector<int> &blockComponentCount) const
562 {
563 size_t export_size = 0;
564 size_t import_size = 0;
565
566 for (size_t bsi = 0; bsi < blockSubsetIndex.size(); bsi++) {
567 int64_t blk_seq = blockSubsetIndex[bsi];
568 const Ioss::BlockDecompositionData &blk = blocks[blk_seq];
569
570 size_t comp_count = blockComponentCount[bsi];
571 export_size += blk.exportMap.size() * comp_count;
572 import_size += blk.importMap.size() * comp_count;
573 }
574
575 std::vector<U> exports;
576 exports.reserve(export_size);
577
578 Ioss::ParallelUtils util_(m_comm);
579 int nProc = util_.parallel_size();
580
581 for (int proc = 0; proc < nProc; proc++) {
582 for (size_t bsi = 0; bsi < blockSubsetIndex.size(); bsi++) {
583 int64_t blk_seq = blockSubsetIndex[bsi];
584 const Ioss::BlockDecompositionData &blk = blocks[blk_seq];
585 size_t comp_count = blockComponentCount[bsi];
586 size_t fileDataOffset = fileOffset[bsi];
587
588 for (int n = 0; n < blk.exportCount[proc]; n++) {
589 int exportIndex = blk.exportIndex[proc] + n;
590 int i = blk.exportMap[exportIndex];
591
592 for (size_t j = 0; j < comp_count; j++) {
593 size_t fileIndex = fileDataOffset + i * comp_count + j;
594 exports.push_back(file_data[fileIndex]);
595 }
596 }
597 }
598 }
599
600 std::vector<int64_t> export_count(nProc, 0);
601 std::vector<int64_t> export_disp(nProc, 0);
602 std::vector<int64_t> import_count(nProc, 0);
603 std::vector<int64_t> import_disp(nProc, 0);
604
605 for (size_t bsi = 0; bsi < blockSubsetIndex.size(); bsi++) {
606 int64_t blk_seq = blockSubsetIndex[bsi];
607 const Ioss::BlockDecompositionData &blk = blocks[blk_seq];
608 size_t comp_count = blockComponentCount[bsi];
609
610 int proc = 0;
611 for (int i : blk.exportCount) {
612 export_count[proc++] += comp_count * i;
613 }
614
615 proc = 0;
616 for (int i : blk.importCount) {
617 import_count[proc++] += comp_count * i;
618 }
619 }
620
621 std::copy(export_count.begin(), export_count.end(), export_disp.begin());
622 std::copy(import_count.begin(), import_count.end(), import_disp.begin());
623
624 Ioss::Utils::generate_index(export_disp);
625 Ioss::Utils::generate_index(import_disp);
626
627 std::vector<U> imports(import_size);
628 Ioss::MY_Alltoallv(exports, export_count, export_disp, imports, import_count, import_disp,
629 m_comm);
630 show_progress("\tCommunication 1 finished");
631
632 Ioss::ElementBlockBatchOffset batchOffset(blocks);
633 std::vector<size_t> iossOffset =
634 batchOffset.get_ioss_offset(blockSubsetIndex, blockComponentCount);
635 std::vector<size_t> importOffset =
636 batchOffset.get_import_offset(blockSubsetIndex, blockComponentCount);
637
638 // Map local and imported data to ioss_data.
639 for (size_t bsi = 0; bsi < blockSubsetIndex.size(); bsi++) {
640 int64_t blk_seq = blockSubsetIndex[bsi];
641 const Ioss::BlockDecompositionData &block = blocks[blk_seq];
642 size_t comp_count = blockComponentCount[bsi];
643
644 for (size_t i = 0; i < block.localMap.size(); i++) {
645 for (size_t j = 0; j < comp_count; j++) {
646 size_t fileIndex = fileOffset[bsi] + block.localMap[i] * comp_count + j;
647 size_t iossIndex = iossOffset[bsi] + (i + block.localIossOffset) * comp_count + j;
648 ioss_data[iossIndex] = file_data[fileIndex];
649 }
650 }
651
652 for (size_t i = 0; i < block.importMap.size(); i++) {
653 for (size_t j = 0; j < comp_count; j++) {
654 size_t importIndex = importOffset[bsi] + i * comp_count + j;
655
656 size_t dataOffset = iossOffset[bsi];
657 size_t iossIndex = dataOffset + block.importMap[i] * comp_count + j;
658
659 ioss_data[iossIndex] = imports[importIndex];
660 }
661 }
662 }
663
664 return iossOffset;
665 }

693 template <typename T>
694 void communicate_node_data(T *file_data, T *ioss_data, size_t comp_count) const
695 {
696 show_progress(__func__);
697 // Transfer the file-decomposition based data in 'file_data' to
698 // the ioss-decomposition based data in 'ioss_data'
699 std::vector<T> export_data(exportNodeMap.size() * comp_count);
700 std::vector<T> import_data(importNodeMap.size() * comp_count);
701
702 if (comp_count == 1) {
703 for (size_t i = 0; i < exportNodeMap.size(); i++) {
704 size_t index = exportNodeMap[i] - m_nodeOffset;
705 assert(index < m_nodeCount);
706 export_data[i] = file_data[index];
707 }
708
709 // Transfer all local data from file_data to ioss_data...
710 for (size_t i = 0; i < localNodeMap.size(); i++) {
711 size_t index = localNodeMap[i] - m_nodeOffset;
712 assert(index < m_nodeCount);
713 ioss_data[m_importPreLocalNodeIndex + i] = file_data[index];
714 }
715
716 // Get my imported data and send my exported data...
717 Ioss::MY_Alltoallv(export_data, exportNodeCount, exportNodeIndex, import_data,
718 importNodeCount, importNodeIndex, m_comm);
719 show_progress("\tCommunication 1a finished");
720
721 // Copy the imported data into ioss_data...
722 for (size_t i = 0; i < importNodeMap.size(); i++) {
723 size_t index = importNodeMap[i];
724 assert(index < ioss_node_count());
725 ioss_data[index] = import_data[i];
726 }
727 }
728 else { // Comp_count > 1
729 for (size_t i = 0; i < exportNodeMap.size(); i++) {
730 size_t index = exportNodeMap[i] - m_nodeOffset;
731 assert(index < m_nodeCount);
732 for (size_t j = 0; j < comp_count; j++) {
733 export_data[comp_count * i + j] = file_data[comp_count * index + j];
734 }
735 }
736
737 // Transfer all local data from file_data to ioss_data...
738 for (size_t i = 0; i < localNodeMap.size(); i++) {
739 size_t index = localNodeMap[i] - m_nodeOffset;
740 assert(index < m_nodeCount);
741 for (size_t j = 0; j < comp_count; j++) {
742 ioss_data[comp_count * (m_importPreLocalNodeIndex + i) + j] =
743 file_data[comp_count * index + j];
744 }
745 }
746
747 std::vector<INT> export_count(exportNodeCount.begin(), exportNodeCount.end());
748 std::vector<INT> export_disp(exportNodeIndex.begin(), exportNodeIndex.end());
749 std::vector<INT> import_count(importNodeCount.begin(), importNodeCount.end());
750 std::vector<INT> import_disp(importNodeIndex.begin(), importNodeIndex.end());
751
752 for (int i = 0; i < m_processorCount; i++) {
753 export_count[i] *= comp_count;
754 export_disp[i] *= comp_count;
755 import_count[i] *= comp_count;
756 import_disp[i] *= comp_count;
757 }
758
759 // Get my imported data and send my exported data...
760 Ioss::MY_Alltoallv(export_data, export_count, export_disp, import_data, import_count,
761 import_disp, m_comm);
762 show_progress("\tCommunication 1b finished");
763
764 // Copy the imported data into ioss_data...
765 for (size_t i = 0; i < importNodeMap.size(); i++) {
766 size_t index = importNodeMap[i];
767 assert(index < ioss_node_count());
768 for (size_t j = 0; j < comp_count; j++) {
769 ioss_data[comp_count * index + j] = import_data[comp_count * i + j];
770 }
771 }
772 }
