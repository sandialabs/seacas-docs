42 using NameTopoKey = std::pair<std::string, const Ioss::ElementTopology *>;
43 struct IONULL_EXPORT NameTopoKeyCompare
44 {
45 bool operator()(const NameTopoKey &lhs, const NameTopoKey &rhs) const
46 {
47 assert(lhs.second != nullptr);
48 assert(rhs.second != nullptr);
49 return lhs.first < rhs.first ||
50 (!(rhs.first < lhs.first) && lhs.second->name() < rhs.second->name());
51 }
52 };
53
54 struct IONULL_EXPORT NameTopoKeyHash
55 {
56 size_t operator()(const NameTopoKey &name_topo) const
57 {
58 return std::hash<std::string>{}(name_topo.first) +
59 std::hash<size_t>{}((size_t)name_topo.second);
60 }
61 };
