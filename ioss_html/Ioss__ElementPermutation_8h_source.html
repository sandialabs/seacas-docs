47 // Permutation data is stored such that the positive permutations are listed first ... the order
48 // of the positive permutations within that group is irrelevant. The remaining permutations listed
49 // after the positive ones are the negative permutations hence, any permutation index outside of
50 // the positive range is a negative permutation. By convention, the first permutation listed
51 // matches the default listed in the Exodus manual

105 // Store low order permutation data regarding this topology .. the positive permutations are
106 // listed first If this topology is a high order topology, the data is only for the nodes of the
107 // associated low order topology. This implies that any usage of this assumes that the higher
108 // order nodes are numbered correctly relative to the low order nodes.
109 //
110 // {{0, 1, 2, 3, 4, 5}, {{0, 1, 2},
111 // {2, 0, 1, 5, 3, 4}, {2, 0, 1},
112 // Tri6 {1, 2, 0, 4, 5, 3}, --> Tri3 {1, 2, 0},
113 // {0, 2, 1, 5, 4, 3}, {0, 2, 1},
114 // {2, 1, 0, 4, 3, 5}, {2, 1, 0},
115 // {1, 0, 2, 3, 5, 4}} {1, 0, 2}}
116
117 void set_permutation(Permutation numPermutationNodes_, Permutation numPermutations_,
118 Permutation numPositivePermutations_,
119 const std::vector<std::vector<Permutation>> &permutationNodeOrdinals_);
120
121 static EPRegistry &registry();
122
123 private:
124 bool equal_(const Ioss::ElementPermutation &rhs, bool quiet) const;

129 Permutation m_numPermutationNodes{0};
