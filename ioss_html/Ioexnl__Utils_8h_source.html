43 using NameTopoKey = std::pair<std::string, const Ioss::ElementTopology *>;
44 struct IOEXNL_EXPORT NameTopoKeyCompare
45 {
46 bool operator()(const NameTopoKey &lhs, const NameTopoKey &rhs) const
47 {
48 assert(lhs.second != nullptr);
49 assert(rhs.second != nullptr);
50 return lhs.first < rhs.first ||
51 (!(rhs.first < lhs.first) && lhs.second->name() < rhs.second->name());
52 }
53 };
54
55 struct IOEXNL_EXPORT NameTopoKeyHash
56 {
57 size_t operator()(const NameTopoKey &name_topo) const
58 {
59 return std::hash<std::string>{}(name_topo.first) +
60 std::hash<size_t>{}((size_t)name_topo.second);
61 }
62 };
