40 using NameTopoKey = std::pair<std::string, const Ioss::ElementTopology *>;
41 struct NameTopoKeyCompare
42 {
43 bool operator()(const NameTopoKey &lhs, const NameTopoKey &rhs) const
44 {
45 assert(lhs.second != nullptr);
46 assert(rhs.second != nullptr);
47 return lhs.first < rhs.first ||
48 (!(rhs.first < lhs.first) && lhs.second->name() < rhs.second->name());
49 }
50 };
51
52 struct NameTopoKeyHash
53 {
54 size_t operator()(const NameTopoKey &name_topo) const
55 {
56 return std::hash<std::string>{}(name_topo.first) +
57 std::hash<size_t>{}((size_t)name_topo.second);
58 }
59 };
