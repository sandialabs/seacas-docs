149 /**
150 * Each bucket entry has:
151 * - A value of type `ValueType`.
152 * - An integer to store how far the value of the bucket, if any, is from its
153 * ideal bucket (ex: if the current bucket 5 has the value 'foo' and
154 * `hash('foo') % nb_buckets` == 3, `dist_from_ideal_bucket()` will return 2 as
155 * the current value of the bucket is two buckets away from its ideal bucket) If
156 * there is no value in the bucket (i.e. `empty()` is true)
157 * `dist_from_ideal_bucket()` will be < 0.
158 * - A marker which tells us if the bucket is the last bucket of the bucket
159 * array (useful for the iterator of the hash table).
160 * - If `StoreHash` is true, 32 bits of the hash of the value, if any, are also
161 * stored in the bucket. If the size of the hash is more than 32 bits, it is
162 * truncated. We don't store the full hash as storing the hash is a potential
163 * opportunity to use the unused space due to the alignment of the bucket_entry
164 * structure. We can thus potentially store the hash without any extra space
165 * (which would not be possible with 64 bits of the hash).
166 */
167 template <typename ValueType, bool StoreHash>
168 class bucket_entry : public bucket_entry_hash<StoreHash>
169 {
170 using bucket_hash = bucket_entry_hash<StoreHash>;

174 using distance_type = std::int16_t;

296 void swap_with_value_in_bucket(distance_type &dist_from_ideal_bucket,
297 truncated_hash_type &my_hash, value_type &value)
298 {
299 tsl_rh_assert(!empty());
300 tsl_rh_assert(dist_from_ideal_bucket > m_dist_from_ideal_bucket);
301
302 using std::swap;
303 swap(value, this->value());
304 swap(dist_from_ideal_bucket, m_dist_from_ideal_bucket);
305
306 if (StoreHash) {
307 const truncated_hash_type tmp_hash = this->truncated_hash();
308 this->set_hash(my_hash);
309 my_hash = tmp_hash;
310 }
311 else {
312 // Avoid warning of unused variable if StoreHash is false
313 TSL_RH_UNUSED(my_hash);
314 }

342 /**
343 * Internal common class used by `robin_map` and `robin_set`.
344 *
345 * ValueType is what will be stored by `robin_hash` (usually `std::pair<Key, T>`
346 * for map and `Key` for set).
347 *
348 * `KeySelect` should be a `FunctionObject` which takes a `ValueType` in
349 * parameter and returns a reference to the key.
350 *
351 * `ValueSelect` should be a `FunctionObject` which takes a `ValueType` in
352 * parameter and returns a reference to the value. `ValueSelect` should be void
353 * if there is no value (in a set for example).
354 *
355 * The strong exception guarantee only holds if the expression
356 * `std::is_nothrow_swappable<ValueType>\:\:value &&
357 * std::is_nothrow_move_constructible<ValueType>\:\:value` is true.
358 *
359 * Behaviour is undefined if the destructor of `ValueType` throws.
360 */
361 template <class ValueType, class KeySelect, class ValueSelect, class Hash, class KeyEqual,
362 class Allocator, bool StoreHash, class GrowthPolicy>
363 class robin_hash : private Hash, private KeyEqual, private GrowthPolicy
364 {
365 private:
366 template <typename U>
367 using has_mapped_type = typename std::integral_constant<bool, !std::is_same<U, void>::value>;
368
369 static_assert(noexcept(std::declval<GrowthPolicy>().bucket_for_hash(std::size_t(0))),
370 "GrowthPolicy::bucket_for_hash must be noexcept.");
371 static_assert(noexcept(std::declval<GrowthPolicy>().clear()),
372 "GrowthPolicy::clear must be noexcept.");

374 public:
375 template <bool IsConst> class robin_iterator;

391 private:
392 /**
393 * Either store the hash because we are asked by the `StoreHash` template
394 * parameter or store the hash because it doesn't cost us anything in size and
395 * can be used to speed up rehash.
396 */
397 static constexpr bool STORE_HASH =
398 StoreHash ||
399 ((sizeof(tsl::detail_robin_hash::bucket_entry<value_type, true>) ==
400 sizeof(tsl::detail_robin_hash::bucket_entry<value_type, false>)) &&
401 (sizeof(std::size_t) == sizeof(truncated_hash_type) ||
402 is_power_of_two_policy<GrowthPolicy>::value) &&
403 // Don't store the hash for primitive types with default hash.
404 (!std::is_arithmetic<key_type>::value || !std::is_same<Hash, std::hash<key_type>>::value));
405
406 /**
407 * Only use the stored hash on lookup if we are explicitly asked. We are not
408 * sure how slow the KeyEqual operation is. An extra comparison may slow
409 * things down with a fast KeyEqual.
410 */
411 static constexpr bool USE_STORED_HASH_ON_LOOKUP = StoreHash;
412
413 /**
414 * We can only use the hash on rehash if the size of the hash type is the same
415 * as the stored one or if we use a power of two modulo. In the case of the
416 * power of two modulo, we just mask the least significant bytes, we just have
417 * to check that the truncated_hash_type didn't truncated more bytes.
418 */
419 static bool USE_STORED_HASH_ON_REHASH(size_type bucket_count)
420 {
421 if (STORE_HASH && sizeof(std::size_t) == sizeof(truncated_hash_type)) {
422 TSL_RH_UNUSED(bucket_count);
423 return true;
424 }
425 else if (STORE_HASH && is_power_of_two_policy<GrowthPolicy>::value) {
426 return bucket_count == 0 ||
427 (bucket_count - 1) <= std::numeric_limits<truncated_hash_type>::max();
428 }
429 else {
430 TSL_RH_UNUSED(bucket_count);
431 return false;
432 }
433 }

797 /**
798 * Here to avoid `template<class K> size_type erase(const K& key)` being used
799 * when we use an `iterator` instead of a `const_iterator`.
800 */
801 iterator erase(iterator pos)
802 {
803 erase_from_bucket(pos);
804
805 /**
806 * Erase bucket used a backward shift after clearing the bucket.
807 * Check if there is a new value in the bucket, if not get the next
808 * non-empty.
809 */
810 if (pos.m_bucket->empty()) {
811 ++pos;
812 }
813
814 m_try_shrink_on_next_insert = true;
815
816 return pos;
817 }

819 iterator erase(const_iterator pos) { return erase(mutable_iterator(pos)); }
820
821 iterator erase(const_iterator first, const_iterator last)
822 {
823 if (first == last) {
824 return mutable_iterator(first);
825 }
826
827 auto first_mutable = mutable_iterator(first);
828 auto last_mutable = mutable_iterator(last);
829 for (auto it = first_mutable.m_bucket; it != last_mutable.m_bucket; ++it) {
830 if (!it->empty()) {
831 it->clear();
832 m_nb_elements--;
833 }
834 }
835
836 if (last_mutable == end()) {
837 m_try_shrink_on_next_insert = true;
838 return end();
839 }
840
841 /*
842 * Backward shift on the values which come after the deleted values.
843 * We try to move the values closer to their ideal bucket.
844 */
845 auto icloser_bucket = static_cast<std::size_t>(first_mutable.m_bucket - m_buckets);
846 auto ito_move_closer_value = static_cast<std::size_t>(last_mutable.m_bucket - m_buckets);
847 tsl_rh_assert(ito_move_closer_value > icloser_bucket);
848
849 const std::size_t ireturn_bucket =
850 ito_move_closer_value -
851 std::min(ito_move_closer_value - icloser_bucket,
852 std::size_t(m_buckets[ito_move_closer_value].dist_from_ideal_bucket()));
853
854 while (ito_move_closer_value < m_bucket_count &&
855 m_buckets[ito_move_closer_value].dist_from_ideal_bucket() > 0) {
856 icloser_bucket =
857 ito_move_closer_value -
858 std::min(ito_move_closer_value - icloser_bucket,
859 std::size_t(m_buckets[ito_move_closer_value].dist_from_ideal_bucket()));
860
861 tsl_rh_assert(m_buckets[icloser_bucket].empty());
862 const distance_type new_distance =
863 distance_type(m_buckets[ito_move_closer_value].dist_from_ideal_bucket() -
864 (ito_move_closer_value - icloser_bucket));
865 m_buckets[icloser_bucket].set_value_of_empty_bucket(
866 new_distance, m_buckets[ito_move_closer_value].truncated_hash(),
867 std::move(m_buckets[ito_move_closer_value].value()));
868 m_buckets[ito_move_closer_value].clear();
869
870 ++icloser_bucket;
871 ++ito_move_closer_value;
872 }
873
874 m_try_shrink_on_next_insert = true;
875
876 return iterator(m_buckets + ireturn_bucket);
877 }

1131 template <class K> const_iterator find_impl(const K &key, std::size_t my_hash) const
1132 {
1133 std::size_t ibucket = bucket_for_hash(my_hash);
1134 distance_type dist_from_ideal_bucket = 0;
1135
1136 while (dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {
1137 if (TSL_RH_LIKELY(
1138 (!USE_STORED_HASH_ON_LOOKUP || m_buckets[ibucket].bucket_hash_equal(my_hash)) &&
1139 compare_keys(KeySelect()(m_buckets[ibucket].value()), key))) {
1140 return const_iterator(m_buckets + ibucket);
1141 }
1142
1143 ibucket = next_bucket(ibucket);
1144 dist_from_ideal_bucket++;
1145 }
1146
1147 return cend();

1150 void erase_from_bucket(iterator pos)
1151 {
1152 pos.m_bucket->clear();
1153 m_nb_elements--;
1154
1155 /**
1156 * Backward shift, swap the empty bucket, previous_ibucket, with the values
1157 * on its right, ibucket, until we cross another empty bucket or if the
1158 * other bucket has a distance_from_ideal_bucket == 0.
1159 *
1160 * We try to move the values closer to their ideal bucket.
1161 */
1162 auto previous_ibucket = static_cast<std::size_t>(pos.m_bucket - m_buckets);
1163 auto ibucket = next_bucket(previous_ibucket);
1164
1165 while (m_buckets[ibucket].dist_from_ideal_bucket() > 0) {
1166 tsl_rh_assert(m_buckets[previous_ibucket].empty());
1167
1168 const auto new_distance = distance_type(m_buckets[ibucket].dist_from_ideal_bucket() - 1);
1169 m_buckets[previous_ibucket].set_value_of_empty_bucket(
1170 new_distance, m_buckets[ibucket].truncated_hash(),
1171 std::move(m_buckets[ibucket].value()));
1172 m_buckets[ibucket].clear();
1173
1174 previous_ibucket = ibucket;
1175 ibucket = next_bucket(ibucket);
1176 }
1177 }

1179 template <class K, class... Args>
1180 std::pair<iterator, bool> insert_impl(const K &key, Args &&...value_type_args)
1181 {
1182 const std::size_t my_hash = hash_key(key);
1183
1184 std::size_t ibucket = bucket_for_hash(my_hash);
1185 distance_type dist_from_ideal_bucket = 0;
1186
1187 while (dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {
1188 if ((!USE_STORED_HASH_ON_LOOKUP || m_buckets[ibucket].bucket_hash_equal(my_hash)) &&
1189 compare_keys(KeySelect()(m_buckets[ibucket].value()), key)) {
1190 return std::make_pair(iterator(m_buckets + ibucket), false);
1191 }
1192
1193 ibucket = next_bucket(ibucket);
1194 dist_from_ideal_bucket++;
1195 }
1196
1197 while (rehash_on_extreme_load(dist_from_ideal_bucket)) {
1198 ibucket = bucket_for_hash(my_hash);
1199 dist_from_ideal_bucket = 0;
1200
1201 while (dist_from_ideal_bucket <= m_buckets[ibucket].dist_from_ideal_bucket()) {
1202 ibucket = next_bucket(ibucket);
1203 dist_from_ideal_bucket++;
1204 }
1205 }
1206
1207 if (m_buckets[ibucket].empty()) {
1208 m_buckets[ibucket].set_value_of_empty_bucket(dist_from_ideal_bucket,
1209 bucket_entry::truncate_hash(my_hash),
1210 std::forward<Args>(value_type_args)...);
1211 }
1212 else {
1213 insert_value(ibucket, dist_from_ideal_bucket, bucket_entry::truncate_hash(my_hash),
1214 std::forward<Args>(value_type_args)...);
1215 }
1216
1217 m_nb_elements++;
1218 /*
1219 * The value will be inserted in ibucket in any case, either because it was
1220 * empty or by stealing the bucket (robin hood).
1221 */
1222 return std::make_pair(iterator(m_buckets + ibucket), true);
1223 }

1234 truncated_hash_type my_hash, value_type &&value)
1235 {
1236 insert_value_impl(ibucket, dist_from_ideal_bucket, my_hash, value);
1237 }

1239 /*
1240 * We don't use `value_type&& value` as last argument due to a bug in MSVC
1241 * when `value_type` is a pointer, The compiler is not able to see the
1242 * difference between `std::string*` and `std::string*&&` resulting in a
1243 * compilation error.
1244 *
1245 * The `value` will be in a moved state at the end of the function.
1246 */
1247 void insert_value_impl(std::size_t ibucket, distance_type dist_from_ideal_bucket,
1248 truncated_hash_type my_hash, value_type &value)
1249 {
1250 tsl_rh_assert(dist_from_ideal_bucket > m_buckets[ibucket].dist_from_ideal_bucket());
1251 m_buckets[ibucket].swap_with_value_in_bucket(dist_from_ideal_bucket, my_hash, value);
1252 ibucket = next_bucket(ibucket);
1253 dist_from_ideal_bucket++;
1254
1255 while (!m_buckets[ibucket].empty()) {
1256 if (dist_from_ideal_bucket > m_buckets[ibucket].dist_from_ideal_bucket()) {
1257 if (dist_from_ideal_bucket > bucket_entry::DIST_FROM_IDEAL_BUCKET_LIMIT) {
1258 /**
1259 * The number of probes is really high, rehash the map on the next
1260 * insert. Difficult to do now as rehash may throw an exception.
1261 */
1262 m_grow_on_next_insert = true;
1263 }
1264
1265 m_buckets[ibucket].swap_with_value_in_bucket(dist_from_ideal_bucket, my_hash, value);
1266 }
1267
1268 ibucket = next_bucket(ibucket);
1269 dist_from_ideal_bucket++;
1270 }
1271
1272 m_buckets[ibucket].set_value_of_empty_bucket(dist_from_ideal_bucket, my_hash,
1273 std::move(value));
1274 }

1276 void rehash_impl(size_type my_count)
1277 {
1278 robin_hash new_table(my_count, static_cast<Hash &>(*this), static_cast<KeyEqual &>(*this),
1279 get_allocator(), m_min_load_factor, m_max_load_factor);
1280
1281 const bool use_stored_hash = USE_STORED_HASH_ON_REHASH(new_table.bucket_count());
1282 for (auto &bucket : m_buckets_data) {
1283 if (bucket.empty()) {
1284 continue;
1285 }
1286
1287 const std::size_t my_hash = use_stored_hash
1288 ? bucket.truncated_hash()
1289 : new_table.hash_key(KeySelect()(bucket.value()));
1290
1291 new_table.insert_value_on_rehash(new_table.bucket_for_hash(my_hash), 0,
1292 bucket_entry::truncate_hash(my_hash),
1293 std::move(bucket.value()));
1294 }
1295
1296 new_table.m_nb_elements = m_nb_elements;
1297 new_table.swap(*this);

1312 void insert_value_on_rehash(std::size_t ibucket, distance_type dist_from_ideal_bucket,
1313 truncated_hash_type my_hash, value_type &&value)
1314 {
1315 while (true) {
1316 if (dist_from_ideal_bucket > m_buckets[ibucket].dist_from_ideal_bucket()) {
1317 if (m_buckets[ibucket].empty()) {
1318 m_buckets[ibucket].set_value_of_empty_bucket(dist_from_ideal_bucket, my_hash,
1319 std::move(value));
1320 return;
1321 }
1322 else {
1323 m_buckets[ibucket].swap_with_value_in_bucket(dist_from_ideal_bucket, my_hash, value);
1324 }
1325 }
1326
1327 dist_from_ideal_bucket++;
1328 ibucket = next_bucket(ibucket);
1329 }
1330 }

1332 /**
1333 * Grow the table if m_grow_on_next_insert is true or we reached the
1334 * max_load_factor. Shrink the table if m_try_shrink_on_next_insert is true
1335 * (an erase occurred) and we're below the min_load_factor.
1336 *
1337 * Return true if the table has been rehashed.
1338 */
1339 bool rehash_on_extreme_load(distance_type curr_dist_from_ideal_bucket)
1340 {
1341 if (m_grow_on_next_insert ||
1342 curr_dist_from_ideal_bucket > bucket_entry::DIST_FROM_IDEAL_BUCKET_LIMIT ||
1343 size() >= m_load_threshold) {
1344 rehash_impl(GrowthPolicy::next_bucket_count());
1345 m_grow_on_next_insert = false;
1346
1347 return true;
1348 }
1349
1350 if (m_try_shrink_on_next_insert) {
1351 m_try_shrink_on_next_insert = false;
1352 if (m_min_load_factor != 0.0f && load_factor() < m_min_load_factor) {
1353 reserve(size() + 1);
1354
1355 return true;
1356 }
1357 }
1358
1359 return false;
1360 }

1362 template <class Serializer> void serialize_impl(Serializer &serializer) const
1363 {
1364 const slz_size_type version = SERIALIZATION_PROTOCOL_VERSION;
1365 serializer(version);
1366
1367 // Indicate if the truncated hash of each bucket is stored. Use a
1368 // std::int16_t instead of a bool to avoid the need for the serializer to
1369 // support an extra 'bool' type.
1370 const auto hash_stored_for_bucket = static_cast<std::int16_t>(STORE_HASH);
1371 serializer(hash_stored_for_bucket);
1372
1373 const slz_size_type nb_elements = m_nb_elements;
1374 serializer(nb_elements);
1375
1376 const slz_size_type bucket_count = m_buckets_data.size();
1377 serializer(bucket_count);
1378
1379 const float min_load_factor = m_min_load_factor;
1380 serializer(min_load_factor);
1381
1382 const float max_load_factor = m_max_load_factor;
1383 serializer(max_load_factor);
1384
1385 for (const bucket_entry &bucket : m_buckets_data) {
1386 if (bucket.empty()) {
1387 const std::int16_t empty_bucket = bucket_entry::EMPTY_MARKER_DIST_FROM_IDEAL_BUCKET;
1388 serializer(empty_bucket);
1389 }
1390 else {
1391 const std::int16_t dist_from_ideal_bucket = bucket.dist_from_ideal_bucket();
1392 serializer(dist_from_ideal_bucket);
1393 if (STORE_HASH) {
1394 const std::uint32_t truncated_hash = bucket.truncated_hash();
1395 serializer(truncated_hash);
1396 }
1397 serializer(bucket.value());
1398 }
1399 }

1402 template <class Deserializer>
1403 void deserialize_impl(Deserializer &deserializer, bool hash_compatible)
1404 {
1405 tsl_rh_assert(m_buckets_data.empty()); // Current hash table must be empty
1406
1407 const slz_size_type version = deserialize_value<slz_size_type>(deserializer);
1408 // For now we only have one version of the serialization protocol.
1409 // If it doesn't match there is a problem with the file.
1410 if (version != SERIALIZATION_PROTOCOL_VERSION) {
1411 TSL_RH_THROW_OR_TERMINATE(std::runtime_error, "Can't deserialize the ordered_map/set. "
1412 "The protocol version header is invalid.");
1413 }
1414
1415 const bool hash_stored_for_bucket =
1416 deserialize_value<std::int16_t>(deserializer) ? true : false;
1417 if (hash_compatible && STORE_HASH != hash_stored_for_bucket) {
1418 TSL_RH_THROW_OR_TERMINATE(std::runtime_error,
1419 "Can't deserialize a map with a different StoreHash "
1420 "than the one used during the serialization when "
1421 "hash compatibility is used");
1422 }
1423
1424 const slz_size_type nb_elements = deserialize_value<slz_size_type>(deserializer);
1425 const slz_size_type bucket_count_ds = deserialize_value<slz_size_type>(deserializer);
1426 const float min_load_factor = deserialize_value<float>(deserializer);
1427 const float max_load_factor = deserialize_value<float>(deserializer);
1428
1429 if (min_load_factor < MINIMUM_MIN_LOAD_FACTOR || min_load_factor > MAXIMUM_MIN_LOAD_FACTOR) {
1430 TSL_RH_THROW_OR_TERMINATE(std::runtime_error,
1431 "Invalid min_load_factor. Check that the serializer "
1432 "and deserializer support floats correctly as they "
1433 "can be converted implicitly to ints.");
1434 }
1435
1436 if (max_load_factor < MINIMUM_MAX_LOAD_FACTOR || max_load_factor > MAXIMUM_MAX_LOAD_FACTOR) {
1437 TSL_RH_THROW_OR_TERMINATE(std::runtime_error,
1438 "Invalid max_load_factor. Check that the serializer "
1439 "and deserializer support floats correctly as they "
1440 "can be converted implicitly to ints.");
1441 }
1442
1443 this->min_load_factor(min_load_factor);
1444 this->max_load_factor(max_load_factor);
1445
1446 if (bucket_count_ds == 0) {
1447 tsl_rh_assert(nb_elements == 0);
1448 return;
1449 }
1450
1451 if (!hash_compatible) {
1452 reserve(numeric_cast<size_type>(nb_elements, "Deserialized nb_elements is too big."));
1453 for (slz_size_type ibucket = 0; ibucket < bucket_count_ds; ibucket++) {
1454 const distance_type dist_from_ideal_bucket =
1455 deserialize_value<std::int16_t>(deserializer);
1456 if (dist_from_ideal_bucket != bucket_entry::EMPTY_MARKER_DIST_FROM_IDEAL_BUCKET) {
1457 if (hash_stored_for_bucket) {
1458 TSL_RH_UNUSED(deserialize_value<std::uint32_t>(deserializer));
1459 }
1460
1461 insert(deserialize_value<value_type>(deserializer));
1462 }
1463 }
1464
1465 tsl_rh_assert(nb_elements == size());
1466 }
1467 else {
1468 m_bucket_count =
1469 numeric_cast<size_type>(bucket_count_ds, "Deserialized bucket_count is too big.");
1470
1471 GrowthPolicy::operator=(GrowthPolicy(m_bucket_count));
1472 // GrowthPolicy should not modify the bucket count we got from
1473 // deserialization
1474 if (m_bucket_count != bucket_count_ds) {
1475 TSL_RH_THROW_OR_TERMINATE(std::runtime_error, "The GrowthPolicy is not the same even "
1476 "though hash_compatible is true.");
1477 }
1478
1479 m_nb_elements =
1480 numeric_cast<size_type>(nb_elements, "Deserialized nb_elements is too big.");
1481 m_buckets_data.resize(m_bucket_count);
1482 m_buckets = m_buckets_data.data();
1483
1484 for (bucket_entry &bucket : m_buckets_data) {
1485 const distance_type dist_from_ideal_bucket =
1486 deserialize_value<std::int16_t>(deserializer);
1487 if (dist_from_ideal_bucket != bucket_entry::EMPTY_MARKER_DIST_FROM_IDEAL_BUCKET) {
1488 truncated_hash_type truncated_hash = 0;
1489 if (hash_stored_for_bucket) {
1490 tsl_rh_assert(hash_stored_for_bucket);
1491 truncated_hash = deserialize_value<std::uint32_t>(deserializer);
1492 }
1493
1494 bucket.set_value_of_empty_bucket(dist_from_ideal_bucket, truncated_hash,
1495 deserialize_value<value_type>(deserializer));
1496 }
1497 }
1498
1499 if (!m_buckets_data.empty()) {
1500 m_buckets_data.back().set_as_last_bucket();
1501 }
1502 }
1503 }
